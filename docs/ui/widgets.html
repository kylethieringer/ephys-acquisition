ui.widgets

Reusable visual widgets for the redesigned GUI.

  • PillToggle: two-option exclusive pill selector (used for Mode and Clamp).
  • StatusBadge: colored LED dot + status text, driven by a state key.
  • TopChromeBar: title bar above the sidebar/workspace showing the current Session ID, Mode/Clamp pills, and the StatusBadge.
  • Sidebar: vertical icon rail that drives a QStackedWidget.

These widgets are pure view — they emit Qt signals but hold no references to acquisition back-ends. ~ui.main_window.MainWindow wires them up.

 37def _make_icon(draw_fn, size: int = 18, color: str = "#b8bec9") -> QIcon:
 38    """Build a QIcon by painting into a transparent pixmap.
 39
 40    Args:
 41        draw_fn: Callable ``(painter, size, pen)`` that draws the glyph.
 42        size: Pixmap edge length in pixels.
 43        color: Stroke color as a hex string.
 44    """
 45    px = QPixmap(size, size)
 46    px.fill(Qt.transparent)
 47    p = QPainter(px)
 48    p.setRenderHint(QPainter.Antialiasing, True)
 49    pen = QPen(QColor(color))
 50    pen.setWidthF(1.6)
 51    pen.setCapStyle(Qt.RoundCap)
 52    pen.setJoinStyle(Qt.RoundJoin)
 53    p.setPen(pen)
 54    draw_fn(p, size, pen)
 55    p.end()
 56    return QIcon(px)

 98def _ic_gear(p: QPainter, s: int, _pen: QPen) -> None:
 99    cx = cy = s / 2
100    p.drawEllipse(int(cx - s * 0.18), int(cy - s * 0.18), int(s * 0.36), int(s * 0.36))
101    for angle in range(0, 360, 45):
102        import math
103        a = math.radians(angle)
104        x1 = cx + math.cos(a) * s * 0.28
105        y1 = cy + math.sin(a) * s * 0.28
106        x2 = cx + math.cos(a) * s * 0.42
107        y2 = cy + math.sin(a) * s * 0.42
108        p.drawLine(int(x1), int(y1), int(x2), int(y2))

124class PillToggle(QFrame):
125    """Two-option exclusive pill selector, styled to match the design's pills.
126
127    Signals:
128        changed(str): Emitted when the selection changes; argument is the
129            value of the newly-selected option.
130    """
131
132    changed = Signal(str)
133
134    def __init__(self, options: list[tuple[str, str]], parent=None) -> None:
135        """
136        Args:
137            options: List of ``(value, label)`` tuples. Exactly two recommended
138                to match the pill visual; more are technically allowed.
139            parent: Qt parent.
140        """
141        super().__init__(parent)
142        self.setProperty("pillbox", True)
143        self.setFrameShape(QFrame.NoFrame)
144
145        layout = QHBoxLayout(self)
146        layout.setContentsMargins(2, 2, 2, 2)
147        layout.setSpacing(2)
148
149        self._group = QButtonGroup(self)
150        self._group.setExclusive(True)
151        self._buttons: dict[str, QPushButton] = {}
152        for value, label in options:
153            btn = QPushButton(label)
154            btn.setCheckable(True)
155            btn.setProperty("pill", True)
156            btn.setCursor(Qt.PointingHandCursor)
157            btn.setSizePolicy(QSizePolicy.Preferred, QSizePolicy.Fixed)
158            btn.clicked.connect(lambda _=False, v=value: self._on_selected(v))
159            layout.addWidget(btn)
160            self._group.addButton(btn)
161            self._buttons[value] = btn
162
163        # Default: first option
164        if options:
165            self._buttons[options[0][0]].setChecked(True)
166
167    def set_value(self, value: str) -> None:
168        """Programmatically select an option without emitting ``changed``."""
169        btn = self._buttons.get(value)
170        if btn and not btn.isChecked():
171            btn.blockSignals(True)
172            btn.setChecked(True)
173            btn.blockSignals(False)
174
175    def value(self) -> str:
176        for v, btn in self._buttons.items():
177            if btn.isChecked():
178                return v
179        return ""
180
181    def _on_selected(self, value: str) -> None:
182        self.changed.emit(value)

208class StatusBadge(QFrame):
209    """LED dot + status text rolled into a single rounded pill.
210
211    Use :meth:`set_state` with a state key (``"idle"``, ``"acquiring"``,
212    ``"recording"``, ``"protocol"``, ``"error"``, ``"stopping"``) to update
213    the color scheme; :meth:`set_text` overrides the label for that state.
214    """
215
216    def __init__(self, parent=None) -> None:
217        super().__init__(parent)
218        self.setFrameShape(QFrame.NoFrame)
219        self._state = "idle"
220
221        layout = QHBoxLayout(self)
222        layout.setContentsMargins(8, 3, 10, 3)
223        layout.setSpacing(7)
224
225        self._led = QFrame()
226        self._led.setFixedSize(QSize(10, 10))
227        self._led.setProperty("led", "idle")
228        layout.addWidget(self._led)
229
230        self._label = QLabel("Ready")
231        self._label.setStyleSheet("font-weight: 600;")
232        layout.addWidget(self._label)
233
234        self.setSizePolicy(QSizePolicy.Maximum, QSizePolicy.Preferred)
235        self._apply()
236
237    def set_state(self, state: str) -> None:
238        """Set the state key and update colors to match.
239
240        Args:
241            state: One of the keys in ``_STATE_STYLES``.  Unknown states
242                default to ``"idle"``.
243        """
244        if state not in _STATE_STYLES:
245            state = "idle"
246        self._state = state
247        self._label.setText(_STATE_STYLES[state].label)
248        self._apply()
249
250    def set_text(self, text: str) -> None:
251        """Override the visible label text for the current state."""
252        self._label.setText(text)
253
254    def _apply(self) -> None:
255        s = _STATE_STYLES[self._state]
256        self._led.setProperty("led", s.led)
257        self._led.style().unpolish(self._led)
258        self._led.style().polish(self._led)
259        self._label.setStyleSheet(f"font-weight: 600; color: {s.fg};")
260        self.setStyleSheet(
261            f"QFrame {{ background-color: {s.bg}; border: 1px solid {s.border};"
262            f" border-radius: 4px; }}"
263        )

270class TopChromeBar(QFrame):
271    """Top chrome row: session label, mode/clamp pills, status badge.
272
273    Widgets are exposed as attributes so :class:`~ui.main_window.MainWindow`
274    can wire them to control-panel signals:
275
276    Attributes:
277        mode_pill: Two-option pill (``"continuous"`` / ``"trial"``).
278        clamp_pill: Two-option pill (``"current_clamp"`` / ``"voltage_clamp"``).
279        status_badge: :class:`StatusBadge` on the right.
280        session_label: Monospace :class:`QLabel` showing the current experiment ID.
281    """
282
283    def __init__(self, parent=None) -> None:
284        super().__init__(parent)
285        self.setProperty("chrome", True)
286        self.setFrameShape(QFrame.NoFrame)
287        self.setFixedHeight(42)
288
289        layout = QHBoxLayout(self)
290        layout.setContentsMargins(14, 4, 14, 4)
291        layout.setSpacing(10)
292
293        # --- Session label ---
294        session_tag = QLabel("SESSION")
295        session_tag.setProperty("tier", "tiny")
296        self.session_label = QLabel("—")
297        self.session_label.setProperty("mono", True)
298        self.session_label.setStyleSheet(
299            "font-weight: 600; color: #e7ebf0; font-size: 10pt;"
300            " font-family: 'JetBrains Mono', 'Consolas', monospace;"
301        )
302        layout.addWidget(session_tag)
303        layout.addWidget(self.session_label)
304
305        layout.addStretch(1)
306
307        # --- Mode pill ---
308        self.mode_pill = PillToggle(
309            [("continuous", "Continuous"), ("trial", "Trial")]
310        )
311        layout.addWidget(self.mode_pill)
312
313        # --- Clamp pill ---
314        self.clamp_pill = PillToggle(
315            [("current_clamp", "CC"), ("voltage_clamp", "VC")]
316        )
317        layout.addWidget(self.clamp_pill)
318
319        # Separator
320        sep = QFrame()
321        sep.setFrameShape(QFrame.VLine)
322        sep.setStyleSheet("color: #3a414c;")
323        sep.setFixedHeight(20)
324        layout.addWidget(sep)
325
326        # --- Status badge ---
327        self.status_badge = StatusBadge()
328        layout.addWidget(self.status_badge)
329
330    def set_session_label(self, text: str) -> None:
331        self.session_label.setText(text or "—")

338class Sidebar(QFrame):
339    """Vertical icon rail that drives a QStackedWidget.
340
341    Signals:
342        page_changed(str): Emitted when a sidebar item is selected; argument
343            is the item ``key``.
344    """
345
346    page_changed = Signal(str)
347
348    def __init__(self, items: list[tuple[str, str, str]], parent=None) -> None:
349        """
350        Args:
351            items: List of ``(key, icon_name, label)`` tuples. ``icon_name``
352                is a key into :data:`ICONS`.
353        """
354        super().__init__(parent)
355        self.setProperty("sidebar", True)
356        self.setFrameShape(QFrame.NoFrame)
357        self.setFixedWidth(84)
358
359        layout = QVBoxLayout(self)
360        layout.setContentsMargins(0, 8, 0, 8)
361        layout.setSpacing(2)
362
363        self._group = QButtonGroup(self)
364        self._group.setExclusive(True)
365        self._buttons: dict[str, QToolButton] = {}
366
367        for key, icon_name, label in items:
368            btn = QToolButton()
369            btn.setText(label)
370            btn.setCheckable(True)
371            btn.setProperty("sidebar", True)
372            btn.setCursor(Qt.PointingHandCursor)
373            btn.setIcon(ICONS.get(icon_name, ICONS["dot"])())
374            btn.setIconSize(QSize(20, 20))
375            btn.setToolButtonStyle(Qt.ToolButtonTextUnderIcon)
376            btn.setFixedHeight(58)
377            btn.setSizePolicy(QSizePolicy.Expanding, QSizePolicy.Fixed)
378            btn.clicked.connect(lambda _=False, k=key: self._on_clicked(k))
379            layout.addWidget(btn)
380            self._group.addButton(btn)
381            self._buttons[key] = btn
382
383        layout.addStretch(1)
384
385        # Default: first item
386        if items:
387            self._buttons[items[0][0]].setChecked(True)
388
389    def set_current(self, key: str) -> None:
390        btn = self._buttons.get(key)
391        if btn:
392            btn.setChecked(True)
393
394    def _on_clicked(self, key: str) -> None:
395        self.page_changed.emit(key)
