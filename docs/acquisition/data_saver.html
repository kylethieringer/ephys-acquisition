acquisition.data_saver

Continuous-mode data saver.

During recording, raw AI data is appended to a flat binary (.bin) file for minimal write overhead. When recording stops, a background BinToHDF5Worker thread converts the binary file to the final HDF5 format. The binary file is always preserved as a raw-data backup.

Binary file format

Consecutive float64 chunks written in C order. Each chunk has shape (N_channels, CHUNK_SIZE) and is written as raw bytes. The full recording is recoverable as::

raw  = np.fromfile(path, dtype=np.float64)
data = (raw.reshape(-1, N_channels, CHUNK_SIZE)
           .transpose(1, 0, 2)
           .reshape(N_channels, -1))

HDF5 file layout

::

/metadata/
    sample_rate      (scalar attribute, int)
    start_time       (scalar attribute, ISO-8601 string)
    channel_names    (string dataset, shape (N_channels,))
    display_scales   (float64 dataset, shape (N_channels,))
    units            (string dataset, shape (N_channels,))
/subject/            (attributes: expt_id, genotype, age, sex, targeted_cell_type)
/data/
    analog_input     (float64, shape (N_channels, N_samples),
                      chunked 1-second blocks, LZF compressed)
/stimulus_events/    (optional — written when events are logged)
    sample_index     (int64 dataset)
    event_type       (string dataset)
    stimulus_name    (string dataset)
    stimulus_index   (int32 dataset)

File naming

{save_dir}/{expt_id}/{expt_id}_{genotype}_{YYYYMMDD}.h5 {save_dir}/{expt_id}/{expt_id}_{genotype}_{YYYYMMDD}.bin

Developer notes

All public methods are called from the GUI thread. ContinuousSaver.close() returns a BinToHDF5Worker. The caller must connect signals and call worker.start(). The .bin file is always kept as a raw-data backup.

 76class BinToHDF5Worker(QThread):
 77    """Background thread that converts a raw binary recording to HDF5.
 78
 79    Reads the ``.bin`` file in :data:`~config.SAMPLE_RATE`-wide column
 80    chunks to keep memory usage bounded.  The binary file is always
 81    preserved as a raw-data backup.
 82
 83    Signals:
 84        conversion_done(str): Emitted with the HDF5 file path string on
 85            successful conversion.
 86        conversion_failed(str): Emitted with an error message string if
 87            conversion raises an exception.
 88    """
 89
 90    conversion_done   = Signal(str)   # HDF5 file path
 91    conversion_failed = Signal(str)   # error message
 92
 93    def __init__(
 94        self,
 95        bin_path: Path,
 96        h5_path:  Path,
 97        header:   dict,
 98        stimulus_events: list | None = None,
 99        parent=None,
100    ) -> None:
101        """
102        Args:
103            bin_path: Path to the source binary file.
104            h5_path: Path where the HDF5 file will be written.
105            header: Metadata dict produced by :meth:`ContinuousSaver.open`.
106            stimulus_events: Optional list of stimulus event dicts (from
107                :meth:`ContinuousSaver.log_event`) to embed under
108                ``/stimulus_events/``.
109        """
110        super().__init__(parent)
111        self._bin_path        = bin_path
112        self._h5_path         = h5_path
113        self._header          = header
114        self._stimulus_events = stimulus_events or []
115
116    def run(self) -> None:
117        """Convert the binary file to HDF5."""
118        try:
119            self._convert()
120            self.conversion_done.emit(str(self._h5_path))
121        except Exception as exc:
122            self.conversion_failed.emit(str(exc))
123
124    def _convert(self) -> None:
125        import h5py as _h5py
126
127        header   = self._header
128        n_ch     = header["n_channels"]
129        bin_size = self._bin_path.stat().st_size
130        n_samp   = bin_size // (n_ch * 8)
131
132        chunk_cols = min(SAMPLE_RATE, max(1, n_samp))
133
134        with open(self._bin_path, "rb") as bf, _h5py.File(self._h5_path, "w") as hf:
135            write_common_hdf5_metadata(hf, header)
136
137            dt_str = _h5py.string_dtype()
138
139            data_grp = hf.create_group("data")
140            ds = data_grp.create_dataset(
141                "analog_input",
142                shape=(n_ch, n_samp),
143                dtype=np.float64,
144                chunks=(n_ch, chunk_cols),
145                compression="lzf",
146            )
147
148            offset          = 0
149            bytes_per_chunk = n_ch * SAMPLE_RATE * 8
150            elems_per_write = n_ch * CHUNK_SIZE
151            while True:
152                raw = bf.read(bytes_per_chunk)
153                if not raw:
154                    break
155                arr  = np.frombuffer(raw, dtype=np.float64)
156                n_wc = len(arr) // elems_per_write
157                if n_wc == 0:
158                    break
159                chunk_data = (
160                    arr[: n_wc * elems_per_write]
161                    .reshape(n_wc, n_ch, CHUNK_SIZE)
162                    .transpose(1, 0, 2)
163                    .reshape(n_ch, n_wc * CHUNK_SIZE)
164                )
165                cols = chunk_data.shape[1]
166                ds[:, offset : offset + cols] = chunk_data
167                offset += cols
168
169            if self._stimulus_events:
170                ev = hf.create_group("stimulus_events")
171                ev.create_dataset(
172                    "sample_index",
173                    data=np.array(
174                        [e["sample_index"] for e in self._stimulus_events],
175                        dtype=np.int64,
176                    ),
177                )
178                ev.create_dataset(
179                    "event_type",
180                    data=np.array(
181                        [e["event_type"] for e in self._stimulus_events],
182                        dtype=object,
183                    ),
184                    dtype=dt_str,
185                )
186                ev.create_dataset(
187                    "stimulus_name",
188                    data=np.array(
189                        [e["stimulus_name"] for e in self._stimulus_events],
190                        dtype=object,
191                    ),
192                    dtype=dt_str,
193                )
194                ev.create_dataset(
195                    "stimulus_index",
196                    data=np.array(
197                        [e["stimulus_index"] for e in self._stimulus_events],
198                        dtype=np.int32,
199                    ),
200                )

207class ContinuousSaver:
208    """Binary-backed recorder for continuous AI data, converted to HDF5 on close.
209
210    During recording, raw float64 data is appended to a ``.bin`` file for
211    minimal I/O overhead.  :meth:`close` returns a :class:`BinToHDF5Worker`
212    that the caller starts after connecting its signals.  The ``.bin`` file
213    is always preserved as a raw-data backup.
214
215    Attributes:
216        CHUNK_COLS (int): HDF5 internal chunk width in samples (1 second).
217    """
218
219    CHUNK_COLS: int = SAMPLE_RATE
220
221    def __init__(self) -> None:
222        self._bin_file:  object        = None
223        self._bin_path:  Path | None   = None
224        self._h5_path:   Path | None   = None
225        self._folder:    Path | None   = None
226        self._header:    dict | None   = None
227        self._n_saved:   int           = 0
228        self._stimulus_events: list    = []
229        self._conversion_worker: "BinToHDF5Worker | None" = None
230
231    # ------------------------------------------------------------------
232    # Public API
233    # ------------------------------------------------------------------
234
235    def open(
236        self,
237        save_dir: str | Path,
238        subject_metadata: dict | None = None,
239        channel_defs=None,
240    ) -> Path:
241        """Open a new binary recording file and store HDF5 metadata.
242
243        Args:
244            save_dir: Root directory for saving.  An experiment subfolder
245                is created automatically.
246            subject_metadata: Optional dict with subject information.
247                Recognised keys: ``"expt_id"``, ``"genotype"``, ``"age"``,
248                ``"sex"``, ``"targeted_cell_type"``.
249            channel_defs: List of :data:`~config.ChannelDef` tuples
250                describing the active AI channels.  Defaults to
251                :data:`~config.AI_CHANNELS` (current-clamp mode).
252
253        Returns:
254            Intended path of the HDF5 file (not yet written; created when
255            the :class:`BinToHDF5Worker` completes).
256
257        Raises:
258            RuntimeError: If h5py is not installed.
259        """
260        if not HAS_H5PY:
261            raise RuntimeError("h5py is not installed. Run: pip install h5py")
262
263        if subject_metadata is None:
264            subject_metadata = {}
265        if channel_defs is None:
266            channel_defs = AI_CHANNELS
267
268        expt_id       = subject_metadata.get("expt_id", "ephys") or "ephys"
269        genotype      = subject_metadata.get("genotype", "") or "unknown"
270        safe_genotype = genotype.replace(" ", "_").replace("/", "-")
271        datestamp     = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
272        stem          = f"{expt_id}_{safe_genotype}_{datestamp}"
273
274        self._folder   = Path(save_dir) / expt_id
275        self._folder.mkdir(parents=True, exist_ok=True)
276        self._h5_path  = self._folder / f"{stem}.h5"
277        self._bin_path = self._folder / f"{stem}.bin"
278
279        self._header = {
280            "sample_rate":    SAMPLE_RATE,
281            "start_time":     datetime.datetime.now().isoformat(),
282            "n_channels":     len(channel_defs),
283            "channel_names":  [ch[0] for ch in channel_defs],
284            "display_scales": [float(ch[3]) for ch in channel_defs],
285            "units":          [ch[4] for ch in channel_defs],
286            "subject": {
287                k: (str(v) if v else "") for k, v in subject_metadata.items()
288            },
289        }
290
291        self._bin_file          = open(self._bin_path, "wb")
292        self._n_saved           = 0
293        self._stimulus_events   = []
294        self._conversion_worker = None
295        return self._h5_path
296
297    def append(self, chunk: NDArray[np.float64]) -> None:
298        """Append a data chunk to the binary file.
299
300        Args:
301            chunk: ``(N_channels, n_samples)`` float64 array in Volts.
302                No-op if the file is not open.
303        """
304        if self._bin_file is None:
305            return
306        n = chunk.shape[1]
307        self._bin_file.write(chunk.astype(np.float64).tobytes())
308        self._n_saved += n
309
310    def log_event(
311        self,
312        sample_idx: int,
313        event_type: str,
314        stim_name: str,
315        stim_idx: int,
316    ) -> None:
317        """Record a stimulus event to be embedded in the HDF5 file.
318
319        Events are collected in memory and written to ``/stimulus_events/``
320        during HDF5 conversion.
321
322        Args:
323            sample_idx: Absolute sample index within the recording when
324                the event occurred.
325            event_type: ``"apply"`` when a waveform is applied,
326                ``"clear"`` when it is removed.
327            stim_name: Human-readable stimulus name.
328            stim_idx: 0-based index into the protocol stimuli list.
329        """
330        self._stimulus_events.append(
331            {
332                "sample_index":   sample_idx,
333                "event_type":     event_type,
334                "stimulus_name":  stim_name,
335                "stimulus_index": stim_idx,
336            }
337        )
338
339    def close(self) -> "BinToHDF5Worker | None":
340        """Close the binary file and return a ready-to-start conversion worker.
341
342        The caller is responsible for connecting the worker's
343        ``conversion_done`` and ``conversion_failed`` signals, then calling
344        ``worker.start()``.
345
346        Returns:
347            A :class:`BinToHDF5Worker` ready to start, or ``None`` if no
348            binary file is present (e.g. :meth:`open` was never called).
349        """
350        if self._bin_file is not None:
351            try:
352                self._bin_file.flush()
353                self._bin_file.close()
354            except Exception:
355                pass
356            finally:
357                self._bin_file = None
358
359        if (
360            self._bin_path is not None
361            and self._bin_path.exists()
362            and self._h5_path is not None
363            and self._header is not None
364        ):
365            worker = BinToHDF5Worker(
366                self._bin_path,
367                self._h5_path,
368                self._header,
369                stimulus_events=list(self._stimulus_events),
370            )
371            self._conversion_worker = worker
372            self._stimulus_events   = []
373            return worker
374
375        return None
376

378    # Properties
379    # ------------------------------------------------------------------
380
381    @property
382    def folder(self) -> Path | None:
383        """Directory containing the recording files, or ``None``."""
384        return self._folder
385
386    @property
387    def is_open(self) -> bool:
388        """``True`` if the binary file is currently open for writing."""
389        return self._bin_file is not None
390
391    @property
392    def path(self) -> Path | None:
393        """Intended HDF5 path (not yet written until conversion completes)."""
394        return self._h5_path
395
396    @property
397    def n_saved(self) -> int:
398        """Total samples written since the last :meth:`open` call."""
399        return self._n_saved
400
401    @property
402    def conversion_worker(self) -> "BinToHDF5Worker | None":
403        """The worker created by the most recent :meth:`close` call."""
404        return self._conversion_worker
