analysis.analysis_gui

Interactive analysis GUI for browsing aligned ephys data and video.

Features:

  • Load an HDF5 recording and auto-locate the corresponding video
  • Synchronized video + ephys trace display with timeline scrubbing
  • Play/pause at adjustable speed
  • Select a time region and export a composite video clip

 161    def set_frame(self, bgr: np.ndarray) -> None:
 162        """Display a BGR (OpenCV) frame."""
 163        if bgr.ndim == 3 and bgr.shape[2] == 3:
 164            rgb = cv2.cvtColor(bgr, cv2.COLOR_BGR2RGB)
 165            img = np.transpose(rgb, (1, 0, 2))
 166        else:
 167            img = bgr.T
 168        max_val = 65535 if img.dtype == np.uint16 else 255
 169        self._image_view.setImage(img, autoRange=self._first_frame,
 170                                  autoLevels=False,
 171                                  levels=(0, max_val),
 172                                  autoHistogramRange=False)
 173        self._first_frame = False
 174
 175    def clear(self) -> None:
 176        self._image_view.setImage(np.zeros((100, 100), dtype=np.uint8))
 177        self._first_frame = True

 289    def load_trace(self, vm: np.ndarray, sr: float,
 290                   frame_samples: np.ndarray) -> None:
 291        """Plot the full Vm trace and store alignment data."""
 292        self._sr = sr
 293        self._frame_samples = frame_samples
 294        t = np.arange(len(vm)) / sr
 295        self._t_max = t[-1]
 296        self._curve.setData(t, vm)
 297
 298        # Set Y spinboxes to data range
 299        y_lo, y_hi = float(np.nanmin(vm)), float(np.nanmax(vm))
 300        margin = (y_hi - y_lo) * 0.08
 301        self._y_min_spin.blockSignals(True)
 302        self._y_max_spin.blockSignals(True)
 303        self._y_min_spin.setValue(y_lo - margin)
 304        self._y_max_spin.setValue(y_hi + margin)
 305        self._y_min_spin.blockSignals(False)
 306        self._y_max_spin.blockSignals(False)
 307
 308        if self._y_auto_cb.isChecked():
 309            self._plot.enableAutoRange(axis="y")
 310        else:
 311            self._apply_y_range()
 312
 313        # Start with full view
 314        self._plot.setXRange(0, t[-1], padding=0.01)
 315        self._region.setRegion([t[-1] * 0.45, t[-1] * 0.55])

 462class ExportWorker(QThread):
 463    """Renders composite video clip in a background thread."""
 464
 465    progress = Signal(int)   # percentage 0-100
 466    finished = Signal(str)   # result message
 467
 468    def __init__(
 469        self,
 470        cap_path: str,
 471        output_path: str,
 472        vm: np.ndarray,
 473        sr: float,
 474        frame_samples: np.ndarray,
 475        start_frame: int,
 476        end_frame: int,
 477        fps: float,
 478        y_lo: float,
 479        y_hi: float,
 480        rmp: float,
 481        brightness: float = 0.0,
 482        contrast: float = 1.0,
 483    ) -> None:
 484        super().__init__()
 485        self._cap_path = cap_path
 486        self._output_path = output_path
 487        self._vm = vm
 488        self._sr = sr
 489        self._frame_samples = frame_samples
 490        self._start = start_frame
 491        self._end = end_frame
 492        self._fps = fps
 493        self._y_lo = y_lo
 494        self._y_hi = y_hi
 495        self._rmp = rmp
 496        self._brightness = brightness
 497        self._contrast = contrast
 498
 499    def run(self) -> None:
 500        cap = cv2.VideoCapture(self._cap_path)
 501        if not cap.isOpened():
 502            self.finished.emit(f"Error: cannot open video {self._cap_path}")
 503            return
 504
 505        vid_w = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
 506        vid_h = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
 507        out_h = vid_h + TRACE_HEIGHT
 508        fourcc = cv2.VideoWriter_fourcc(*"MJPG")
 509        writer = cv2.VideoWriter(self._output_path, fourcc, self._fps,
 510                                 (vid_w, out_h))
 511        if not writer.isOpened():
 512            cap.release()
 513            self.finished.emit(f"Error: cannot write to {self._output_path}")
 514            return
 515
 516        n_clip = self._end - self._start
 517        cap.set(cv2.CAP_PROP_POS_FRAMES, self._start)
 518
 519        for i in range(n_clip):
 520            ret, frame = cap.read()
 521            if not ret:
 522                break
 523
 524            frame_idx = self._start + i
 525            center_sample = int(self._frame_samples[frame_idx])
 526            trace_img = _draw_trace(
 527                self._vm, center_sample, self._sr, vid_w,
 528                self._y_lo, self._y_hi, self._rmp,
 529            )
 530
 531            if frame.shape[1] != vid_w:
 532                frame = cv2.resize(frame, (vid_w, vid_h))
 533
 534            if self._contrast != 1.0 or self._brightness != 0.0:
 535                frame = cv2.convertScaleAbs(
 536                    frame, alpha=self._contrast, beta=self._brightness,
 537                )
 538
 539            composite = np.vstack([frame, trace_img])
 540            writer.write(composite)
 541
 542            pct = int(100 * (i + 1) / n_clip)
 543            self.progress.emit(pct)
 544
 545        cap.release()
 546        writer.release()
 547        self.finished.emit(f"Saved clip to {self._output_path}")

 661    def _auto_adjust(self) -> None:
 662        """Sample frames from the clip range and pick brightness/contrast
 663        that map the 1st–99th percentile of luminance to 0–255."""
 664        cap = cv2.VideoCapture(self._video_path)
 665        if not cap.isOpened():
 666            return
 667
 668        n_samples = 10
 669        n_clip = max(1, self._end_frame - self._start_frame)
 670        step = max(1, n_clip // n_samples)
 671        indices = list(range(self._start_frame, self._end_frame, step))[:n_samples]
 672
 673        pixels = []
 674        for idx in indices:
 675            cap.set(cv2.CAP_PROP_POS_FRAMES, idx)
 676            ret, frame = cap.read()
 677            if not ret:
 678                continue
 679            if frame.ndim == 3:
 680                frame = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
 681            pixels.append(frame.ravel())
 682        cap.release()
 683
 684        if not pixels:
 685            return
 686
 687        data = np.concatenate(pixels)
 688        p_lo, p_hi = np.percentile(data, [1.0, 99.0])
 689        if p_hi - p_lo < 1.0:
 690            return
 691
 692        alpha = 255.0 / (p_hi - p_lo)
 693        beta = -alpha * p_lo
 694        alpha = float(np.clip(alpha, 0.1, 5.0))
 695        beta = float(np.clip(beta, -255.0, 255.0))
 696        self._contrast_spin.setValue(alpha)
 697        self._brightness_spin.setValue(beta)

 937    def _find_video(self, h5_path: Path,
 938                    trial_num: int | None) -> Path | None:
 939        """Try to auto-locate the video for the given HDF5 file."""
 940        directory = h5_path.parent
 941
 942        # 1) Check metadata JSON sidecar
 943        meta_path = directory / (h5_path.stem + "_metadata.json")
 944        if meta_path.exists():
 945            try:
 946                with open(meta_path) as f:
 947                    meta = json.load(f)
 948                vid_name = meta.get("files", {}).get("video")
 949                if vid_name:
 950                    vid_path = directory / vid_name
 951                    if vid_path.exists():
 952                        return vid_path
 953            except (json.JSONDecodeError, KeyError):
 954                pass
 955
 956        # 2) Trial mode: check video_file attribute
 957        if trial_num is not None:
 958            try:
 959                with h5py.File(h5_path, "r") as f:
 960                    key = f"trial_{trial_num:03d}"
 961                    vid_name = f[key].attrs.get("video_file", "")
 962                    if vid_name:
 963                        vid_path = directory / vid_name
 964                        if vid_path.exists():
 965                            return vid_path
 966            except (KeyError, OSError):
 967                pass
 968
 969        # 3) Look for .avi with matching stem in the same directory
 970        for avi in sorted(directory.glob("*.avi")):
 971            if avi.stem.startswith(h5_path.stem.replace("_metadata", "")):
 972                return avi
 973
 974        return None

 978    def _seek_to_frame(self, frame_idx: int) -> None:
 979        if self._frame_samples is None or self._n_frames == 0:
 980            return
 981        if self._seeking:
 982            return
 983        self._seeking = True
 984        try:
 985            frame_idx = max(0, min(frame_idx, self._n_frames - 1))
 986            sample_idx = int(self._frame_samples[frame_idx])
 987            time_s = sample_idx / self._sr
 988
 989            # Video
 990            if self._cap is not None:
 991                # Sequential read if next frame, otherwise seek
 992                if frame_idx != self._current_frame + 1:
 993                    self._cap.set(cv2.CAP_PROP_POS_FRAMES, frame_idx)
 994                ret, frame = self._cap.read()
 995                if ret:
 996                    self._video_panel.set_frame(frame)
 997
 998            # Trace cursor
 999            self._trace_panel.set_cursor(time_s)
1000
1001            # Timeline
1002            self._timeline.set_position(frame_idx)
1003            self._timeline.update_labels(frame_idx, self._n_frames, time_s)
1004
1005            self._current_frame = frame_idx
1006        finally:
1007            self._seeking = False
