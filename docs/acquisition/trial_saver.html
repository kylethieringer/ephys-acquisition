acquisition.trial_saver

Trial-based data saver.

During a protocol run, each trial's AI data is appended to a flat binary (.bin) file as it arrives. An in-memory trial index table tracks the byte offset and sample count of every trial. When the protocol ends, TrialSaver.close() converts the binary file to the final per-trial HDF5 layout. The binary file is always preserved as a raw-data backup.

Binary file format

Consecutive float64 trial blocks written in C order::

trial_0_block  (N_channels × n_trial_samples_0, float64)
trial_1_block  (N_channels × n_trial_samples_1, float64)
...

Each block is recoverable as::

raw   = np.frombuffer(raw_bytes, dtype=np.float64)
data  = raw.reshape(N_channels, n_samples)

HDF5 file layout (unchanged schema)

::

/metadata/
    sample_rate            (scalar attribute, int)
    start_time             (scalar attribute, ISO-8601 string)
    clamp_mode             (scalar attribute, str)
    n_trials               (scalar attribute, int)
    channel_names          (string dataset)
    display_scales         (float64 dataset)
    units                  (string dataset)
    protocol               (string dataset — full JSON)
    trial_order            (int32 dataset)
    trial_stimulus_names   (string dataset)
    trial_stimulus_types   (string dataset)
/subject/                  (attributes)
/trial_001/
    analog_input           (float64, shape (N_channels, N_samples))
    attrs: stimulus_name, stimulus_index, trial_index, onset_time, video_file

File naming

{save_dir}/{expt_id}/{expt_id}_{genotype}_{YYYYMMDD_HHMMSS}_trials.h5 {save_dir}/{expt_id}/{expt_id}_{genotype}_{YYYYMMDD_HHMMSS}_trials.bin

Developer notes

All public methods must be called from the GUI thread. TrialSaver.close() converts .bin → HDF5 synchronously on the GUI thread (trial files are small, conversion is fast). The .bin file is always kept as a raw-data backup.

 78    """Binary-backed writer for per-trial data, converted to HDF5 on close.
 79
 80    Trials are appended to a ``.bin`` file during the run.  An in-memory

 87        _path (Path | None): Intended HDF5 file path.
 88        _folder (Path | None): Directory containing the files.
 89        _trial_index (list[dict]): In-memory per-trial metadata table.
 90    """

 93        self._bin_file:   object        = None
 94        self._bin_path:   Path | None   = None
 95        self._path:       Path | None   = None
 96        self._folder:     Path | None   = None
 97        self._header:     dict | None   = None
 98        self._trial_index: list[dict]   = []

104    def open(
105        self,
106        save_dir: str | Path,
107        protocol: TrialProtocol,
108        trial_order: list[int],
109        subject_metadata: dict,
110        channel_defs: list[ChannelDef],
111    ) -> Path:
112        """Open a new binary file and store HDF5 metadata for later conversion.
113
114        Args:
115            save_dir: Root directory for saving.
116            protocol: The :class:`~acquisition.trial_protocol.TrialProtocol`
117                being run.
118            trial_order: Shuffled list of stimulus indices.
119            subject_metadata: Dict with subject information.
120            channel_defs: List of :data:`~config.ChannelDef` tuples for
121                the active channels (CC or VC).
122
123        Returns:
124            Intended path of the HDF5 file (not yet written).
125
126        Raises:
127            RuntimeError: If h5py is not installed.
128        """
129        if not HAS_H5PY:
130            raise RuntimeError("h5py is not installed. Run: pip install h5py")
131
132        expt_id       = subject_metadata.get("expt_id", "ephys") or "ephys"
133        genotype      = subject_metadata.get("genotype", "") or "unknown"
134        safe_genotype = genotype.replace(" ", "_").replace("/", "-")
135        datestamp     = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
136        stem          = f"{expt_id}_{safe_genotype}_{datestamp}_trials"
137
138        self._folder   = Path(save_dir) / expt_id
139        self._folder.mkdir(parents=True, exist_ok=True)
140        self._path     = self._folder / f"{stem}.h5"
141        self._bin_path = self._folder / f"{stem}.bin"
142
143        self._header = {
144            "sample_rate":    SAMPLE_RATE,
145            "start_time":     datetime.datetime.now().isoformat(),
146            "n_channels":     len(channel_defs),
147            "channel_names":  [ch[0] for ch in channel_defs],
148            "display_scales": [float(ch[3]) for ch in channel_defs],
149            "units":          [ch[4] for ch in channel_defs],
150            "clamp_mode":     protocol.clamp_mode,
151            "n_trials":       len(trial_order),
152            "protocol_json":  json.dumps(protocol_to_dict(protocol), indent=2),
153            "trial_order":    list(trial_order),
154            "trial_stimulus_names": [protocol.stimuli[i].name for i in trial_order],
155            "trial_stimulus_types": [protocol.stimuli[i].type for i in trial_order],
156            "subject": {
157                k: (str(v) if v else "") for k, v in subject_metadata.items()
158            },
159        }
160
161        self._bin_file    = open(self._bin_path, "wb")
162        self._trial_index = []
163        return self._path

165    def write_trial(
166        self,
167        trial_index:    int,
168        stimulus_index: int,
169        stimulus_name:  str,
170        onset_time:     str,
171        data:           NDArray[np.float64],
172        video_filename: str = "",
173    ) -> None:
174        """Append one trial's data to the binary file and record its metadata.
175
176        Args:
177            trial_index: 0-based index of this trial in the run.
178            stimulus_index: 0-based index into ``protocol.stimuli``.
179            stimulus_name: Human-readable stimulus label.
180            onset_time: ISO-8601 timestamp string for the trial onset.
181            data: ``(N_channels, n_samples)`` float64 array in Volts.
182            video_filename: Basename of the ``.avi`` file for this trial.
183                Empty string if no video was saved.
184        """
185        if self._bin_file is None:
186            return
187
188        n_samples    = data.shape[1]
189        byte_offset  = self._bin_file.tell()
190        self._bin_file.write(data.astype(np.float64).tobytes())
191        self._bin_file.flush()
192
193        self._trial_index.append(
194            {
195                "trial_index":    trial_index,
196                "stimulus_index": stimulus_index,
197                "stimulus_name":  stimulus_name,
198                "onset_time":     onset_time,
199                "n_samples":      n_samples,
200                "byte_offset":    byte_offset,
201                "video_filename": video_filename,
202            }
203        )
204
205    def close(self) -> Path | None:
206        """Convert the binary file to HDF5 and return the HDF5 path.
207
208        Converts synchronously on the calling thread (GUI thread) — trial
209        files are small enough that this is acceptable.  The ``.bin`` file
210        is always preserved as a backup.
211
212        Returns:
213            Full path to the HDF5 file, or ``None`` if the file was
214            never opened.
215        """
216        if self._bin_file is not None:
217            try:
218                self._bin_file.flush()
219                self._bin_file.close()
220            except Exception:
221                pass
222            finally:
223                self._bin_file = None
224
225        if (
226            self._bin_path is None
227            or not self._bin_path.exists()
228            or self._header is None
229            or self._path is None
230        ):
231            return self._path
232
233        try:
234            self._convert_to_hdf5()
235        except Exception:
236            pass  # keep .bin; HDF5 may be incomplete but bin is intact
237
238        return self._path

263    def _convert_to_hdf5(self) -> None:
264        """Read the binary file and write the per-trial HDF5 layout."""
265        import h5py as _h5py
266
267        header = self._header
268        n_ch   = header["n_channels"]
269
270        with open(self._bin_path, "rb") as bf, _h5py.File(self._path, "w") as hf:
271            dt_str = _h5py.string_dtype()
272
273            # /metadata and /subject (common fields)
274            meta = write_common_hdf5_metadata(hf, header)
275
276            # Trial-specific metadata
277            meta.attrs["clamp_mode"] = header["clamp_mode"]
278            meta.attrs["n_trials"]   = header["n_trials"]
279            meta.create_dataset("protocol", data=header["protocol_json"], dtype=dt_str)
280            meta.create_dataset(
281                "trial_order",
282                data=np.array(header["trial_order"], dtype=np.int32),
283            )
284            meta.create_dataset(
285                "trial_stimulus_names",
286                data=np.array(header["trial_stimulus_names"], dtype=object),
287                dtype=dt_str,
288            )
289            meta.create_dataset(
290                "trial_stimulus_types",
291                data=np.array(header["trial_stimulus_types"], dtype=object),
292                dtype=dt_str,
293            )
294
295            # /trial_NNN groups
296            for entry in self._trial_index:
297                group_name = f"trial_{entry['trial_index'] + 1:03d}"
298                n_samp     = entry["n_samples"]
299
300                bf.seek(entry["byte_offset"])
301                raw  = bf.read(n_ch * n_samp * 8)
302                data = np.frombuffer(raw, dtype=np.float64).reshape(n_ch, n_samp)
303
304                grp = hf.create_group(group_name)
305                grp.attrs["stimulus_name"]  = entry["stimulus_name"]
306                grp.attrs["stimulus_index"] = entry["stimulus_index"]
307                grp.attrs["trial_index"]    = entry["trial_index"]
308                grp.attrs["onset_time"]     = entry["onset_time"]
309                grp.attrs["video_file"]     = entry["video_filename"]
310                grp.create_dataset("analog_input", data=data, dtype=np.float64)
