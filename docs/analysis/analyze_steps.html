analysis.analyze_steps

Interactive analysis of current-injection staircase recordings.

Loads a continuous-mode HDF5 recording, detects staircase stimuli (using protocol metadata when available), computes intrinsic cell properties (resting membrane potential and input resistance), and lets you interactively browse and save overlay plots of the step responses.

Usage::

python analyze_steps.py

A file-picker dialog will open to select the HDF5 file.

 103def load_continuous_h5(h5_path: Path) -> dict:
 104    """Load a continuous-mode HDF5 recording.
 105
 106    Parameters
 107    ----------
 108    h5_path : Path
 109        Path to the ``.h5`` file (must contain ``/data/analog_input``).
 110
 111    Returns
 112    -------
 113    dict
 114        Keys: ``data`` (n_channels x n_samples ndarray), ``sample_rate``,
 115        ``channel_names``, ``display_scales``, ``units``,
 116        ``stimulus_events`` (dict or None), ``subject`` (dict).
 117
 118    Raises
 119    ------
 120    ValueError
 121        If the file is trial-based rather than continuous.
 122    """
 123    result = {}
 124    with h5py.File(h5_path, "r") as f:
 125        # Reject trial-based files
 126        trial_keys = [k for k in f.keys() if k.startswith("trial_")]
 127        if trial_keys:
 128            raise ValueError(
 129                "This is a trial-based file. Only continuous recordings "
 130                "are supported."
 131            )
 132
 133        # Metadata
 134        meta = f["metadata"]
 135        result["sample_rate"] = int(meta.attrs["sample_rate"])
 136        result["channel_names"] = [
 137            s.decode() if isinstance(s, bytes) else s
 138            for s in meta["channel_names"][:]
 139        ]
 140        result["display_scales"] = meta["display_scales"][:]
 141        result["units"] = [
 142            s.decode() if isinstance(s, bytes) else s
 143            for s in meta["units"][:]
 144        ]
 145
 146        # Raw analog data
 147        result["data"] = f["data/analog_input"][:]
 148
 149        # Stimulus events (optional)
 150        if "stimulus_events" in f:
 151            ev = f["stimulus_events"]
 152            result["stimulus_events"] = {
 153                "sample_index": ev["sample_index"][:],
 154                "event_type": [
 155                    s.decode() if isinstance(s, bytes) else s
 156                    for s in ev["event_type"][:]
 157                ],
 158                "stimulus_name": [
 159                    s.decode() if isinstance(s, bytes) else s
 160                    for s in ev["stimulus_name"][:]
 161                ],
 162                "stimulus_index": ev["stimulus_index"][:],
 163            }
 164        else:
 165            result["stimulus_events"] = None
 166
 167        # Subject metadata
 168        if "subject" in f:
 169            result["subject"] = dict(f["subject"].attrs)
 170        else:
 171            result["subject"] = {}
 172
 173    return result

 176def load_protocol_metadata(h5_path: Path) -> list[dict]:
 177    """Load protocol run(s) from the sidecar ``_metadata.json`` file.
 178
 179    Each entry in the returned list has ``start_sample`` (int) and
 180    ``protocol`` (a ``TrialProtocol`` instance parsed from JSON).
 181
 182    Parameters
 183    ----------
 184    h5_path : Path
 185        Path to the ``.h5`` file; the sidecar is assumed to be at
 186        ``<stem>_metadata.json`` in the same directory.
 187
 188    Returns
 189    -------
 190    list[dict]
 191        One dict per protocol run with keys ``start_sample`` and
 192        ``protocol``.  Empty list if no sidecar or no protocol entries.
 193    """
 194    meta_path = h5_path.with_name(h5_path.stem + "_metadata.json")
 195    if not meta_path.exists():
 196        return []
 197    metadata = json.loads(meta_path.read_text())
 198    runs = []
 199    for entry in metadata.get("protocols", []) or []:
 200        proto_dict = entry.get("protocol")
 201        if proto_dict is None:
 202            continue
 203        runs.append({
 204            "start_sample": entry.get("start_sample", 0),
 205            "protocol": protocol_from_dict(proto_dict),
 206            "protocol_dict": proto_dict,
 207        })
 208    return runs

 215def smooth_cmd(data: np.ndarray, sr: int, display_scales: np.ndarray,
 216               cmd_ch: int = CMD_CH) -> np.ndarray:
 217    """Median-filter the command-current channel and scale to pA.
 218
 219    A 1 ms median-filter kernel removes fast transition artifacts while
 220    preserving the flat plateaus of each current step.
 221
 222    Parameters
 223    ----------
 224    data : ndarray, shape (n_channels, n_samples)
 225        Raw recording data.
 226    sr : int
 227        Sampling rate in Hz.
 228    display_scales : ndarray
 229        Per-channel scale factors (raw volts -> display units).
 230    cmd_ch : int
 231        Index of the AmpCmd channel.
 232
 233    Returns
 234    -------
 235    ndarray, shape (n_samples,)
 236        Smoothed command current in pA.
 237    """
 238    kernel = max(3, ms_to_samples(1.0, sr) | 1)  # odd, >= 3
 239    return median_filter(data[cmd_ch] * display_scales[cmd_ch], size=kernel)

 242def find_pulses(cmd_pA: np.ndarray, sr: int,
 243                min_amp_pA: float = MIN_AMP_PA,
 244                min_step_ms: float = MIN_STEP_MS) -> list[dict]:
 245    """Find contiguous current-injection pulses above a threshold.
 246
 247    Detects every run of samples where ``|cmd| >= min_amp_pA`` that lasts
 248    at least ``min_step_ms``.  The amplitude of each pulse is the median
 249    of the smoothed command signal during the run.
 250
 251    Parameters
 252    ----------
 253    cmd_pA : ndarray
 254        Smoothed command-current waveform in pA (1-D).
 255    sr : int
 256        Sampling rate in Hz.
 257    min_amp_pA : float
 258        Minimum absolute amplitude to consider (pA).
 259    min_step_ms : float
 260        Minimum duration for a valid pulse (ms).
 261
 262    Returns
 263    -------
 264    list[dict]
 265        Each dict has ``onset`` (int), ``offset`` (int), and
 266        ``amplitude_pA`` (float).  Indices are relative to *cmd_pA*.
 267    """
 268    min_samples = ms_to_samples(min_step_ms, sr)
 269    above = np.abs(cmd_pA) >= min_amp_pA
 270
 271    ab = above.astype(np.int8)
 272    edges = np.diff(ab)
 273    rising = np.flatnonzero(edges == 1) + 1
 274    falling = np.flatnonzero(edges == -1) + 1
 275    if above[0]:
 276        rising = np.concatenate(([0], rising))
 277    if above[-1]:
 278        falling = np.concatenate((falling, [len(above)]))
 279
 280    pulses = []
 281    for onset, offset in zip(rising, falling):
 282        if offset - onset < min_samples:
 283            continue
 284        amp = float(np.median(cmd_pA[onset:offset]))
 285        pulses.append({
 286            "onset": int(onset),
 287            "offset": int(offset),
 288            "amplitude_pA": amp,
 289        })
 290    return pulses

 297def _find_protocol_run(apply_sample: int,
 298                       protocol_runs: list[dict]) -> dict | None:
 299    """Return the protocol run whose start_sample is closest to (and <=)
 300    *apply_sample*."""
 301    best = None
 302    for run in protocol_runs:
 303        start = run.get("start_sample", 0)
 304        if start <= apply_sample:
 305            if best is None or start > best.get("start_sample", 0):
 306                best = run
 307    return best

 310def _expected_step_count(stim_def: dict,
 311                         min_amp_pA: float = MIN_AMP_PA) -> int | None:
 312    """Predict the detectable pulse count for a staircase stimulus.
 313
 314    Parameters
 315    ----------
 316    stim_def : dict
 317        Raw stimulus definition dict (from protocol JSON).
 318    min_amp_pA : float
 319        Detection threshold — amplitudes below this are invisible.
 320
 321    Returns
 322    -------
 323    int or None
 324        Expected number of pulses, or None if not a staircase.
 325    """
 326    if stim_def.get("type") != "staircase":
 327        return None
 328    min_pA = stim_def.get("min_pA")
 329    max_pA = stim_def.get("max_pA")
 330    step_pA = stim_def.get("step_pA")
 331    repeats = stim_def.get("staircase_repeats") or 1
 332    if None in (min_pA, max_pA, step_pA) or step_pA <= 0:
 333        return None
 334    n = 0
 335    val = min_pA
 336    while val <= max_pA + 1e-9:
 337        if abs(val) >= min_amp_pA:
 338            n += 1
 339        val += step_pA
 340    return n * repeats

 343def find_staircases_from_events(
 344    data: np.ndarray,
 345    stimulus_events: dict,
 346    protocol_runs: list[dict],
 347    sr: int,
 348    display_scales: np.ndarray,
 349) -> tuple[list[list[dict]], list[dict]]:
 350    """Detect one staircase per apply->clear stimulus-event window.
 351
 352    Pulses are found within each window by thresholding the smoothed AmpCmd
 353    channel.  When protocol metadata is available, the expected step count
 354    is included in the per-staircase metadata for validation.
 355
 356    Parameters
 357    ----------
 358    data : ndarray, shape (n_channels, n_samples)
 359        Full raw recording.
 360    stimulus_events : dict
 361        Must contain ``sample_index``, ``event_type``, ``stimulus_name``,
 362        ``stimulus_index`` arrays.
 363    protocol_runs : list[dict]
 364        Parsed protocol runs from the sidecar JSON.
 365    sr : int
 366        Sampling rate in Hz.
 367    display_scales : ndarray
 368        Per-channel scale factors.
 369
 370    Returns
 371    -------
 372    staircases : list[list[dict]]
 373        Each staircase is a list of pulse dicts (``onset``, ``offset``,
 374        ``amplitude_pA``) with sample indices into *data*.
 375    metadata : list[dict]
 376        Per-staircase metadata: ``apply_sample``, ``clear_sample``,
 377        ``stimulus_name``, ``stimulus_index``, ``expected_steps``,
 378        ``found_steps``.
 379    """
 380    cmd_smooth = smooth_cmd(data, sr, display_scales)
 381    n_total = data.shape[1]
 382
 383    ev_types = stimulus_events["event_type"]
 384    ev_samples = stimulus_events["sample_index"]
 385    ev_names = stimulus_events["stimulus_name"]
 386    ev_idx = stimulus_events["stimulus_index"]
 387    n_ev = len(ev_types)
 388
 389    staircases: list[list[dict]] = []
 390    meta: list[dict] = []
 391
 392    for i, etype in enumerate(ev_types):
 393        if etype != "apply":
 394            continue
 395        start_sample = int(ev_samples[i])
 396
 397        # Find the matching clear (or next apply) to define the window
 398        end_sample = n_total
 399        for j in range(i + 1, n_ev):
 400            if ev_types[j] in ("clear", "apply"):
 401                end_sample = int(ev_samples[j])
 402                break
 403
 404        start_sample = max(0, min(start_sample, n_total))
 405        end_sample = max(start_sample, min(end_sample, n_total))
 406        if end_sample <= start_sample:
 407            continue
 408
 409        pulses = find_pulses(cmd_smooth[start_sample:end_sample], sr)
 410        if not pulses:
 411            continue
 412
 413        # Shift indices to be relative to the full recording
 414        steps = [
 415            {
 416                "onset": p["onset"] + start_sample,
 417                "offset": p["offset"] + start_sample,
 418                "amplitude_pA": p["amplitude_pA"],
 419            }
 420            for p in pulses
 421        ]
 422
 423        # Check expected step count from protocol if available
 424        expected = None
 425        protocol_runs_raw = [
 426            r.get("protocol_dict") for r in protocol_runs if r.get("protocol_dict")
 427        ]
 428        if protocol_runs:
 429            run = _find_protocol_run(start_sample, protocol_runs)
 430            if run is not None:
 431                stim_list = run.get("protocol_dict", {}).get("stimuli", [])
 432                sidx = int(ev_idx[i])
 433                if 0 <= sidx < len(stim_list):
 434                    expected = _expected_step_count(stim_list[sidx])
 435
 436        staircases.append(steps)
 437        meta.append({
 438            "apply_sample": start_sample,
 439            "clear_sample": end_sample,
 440            "stimulus_name": ev_names[i],
 441            "stimulus_index": int(ev_idx[i]),
 442            "expected_steps": expected,
 443            "found_steps": len(steps),
 444        })
 445
 446    return staircases, meta

 449def find_staircases_from_waveform(
 450    data: np.ndarray,
 451    sr: int,
 452    display_scales: np.ndarray,
 453    max_gap_ms: float = MAX_GAP_MS,
 454) -> tuple[list[list[dict]], list[dict]]:
 455    """Fallback staircase detection from the AmpCmd waveform alone.
 456
 457    Detects all pulses in the full recording and groups them into
 458    staircases separated by silences longer than *max_gap_ms*.
 459
 460    Parameters
 461    ----------
 462    data : ndarray, shape (n_channels, n_samples)
 463        Full raw recording.
 464    sr : int
 465        Sampling rate in Hz.
 466    display_scales : ndarray
 467        Per-channel scale factors.
 468    max_gap_ms : float
 469        Maximum silence (ms) between consecutive steps in the same
 470        staircase.
 471
 472    Returns
 473    -------
 474    staircases : list[list[dict]]
 475        Grouped pulse lists.
 476    metadata : list[dict]
 477        Minimal metadata (no stimulus names or expected counts).
 478    """
 479    cmd_smooth = smooth_cmd(data, sr, display_scales)
 480    all_pulses = find_pulses(cmd_smooth, sr)
 481
 482    max_gap_samples = ms_to_samples(max_gap_ms, sr)
 483    if not all_pulses:
 484        return [], []
 485
 486    staircases: list[list[dict]] = [[all_pulses[0]]]
 487    for prev, pulse in zip(all_pulses[:-1], all_pulses[1:]):
 488        if pulse["onset"] - prev["offset"] > max_gap_samples:
 489            staircases.append([pulse])
 490        else:
 491            staircases[-1].append(pulse)
 492
 493    meta = [
 494        {
 495            "apply_sample": sc[0]["onset"],
 496            "clear_sample": sc[-1]["offset"],
 497            "stimulus_name": "unknown",
 498            "stimulus_index": -1,
 499            "expected_steps": None,
 500            "found_steps": len(sc),
 501        }
 502        for sc in staircases
 503    ]
 504    return staircases, meta

 511def separate_hyperpol(
 512    pulses: list[dict],
 513    protocol_runs: list[dict],
 514    apply_sample: int,
 515    stimulus_index: int,
 516) -> tuple[dict | None, list[dict]]:
 517    """Separate the hyperpolarization pulse from staircase steps.
 518
 519    The hyperpolarization pulse is the first negative-amplitude pulse in
 520    the window, provided the protocol metadata confirms one should exist.
 521    If no protocol metadata is available, the first negative pulse is
 522    still treated as hyperpol if it precedes ascending staircase steps.
 523
 524    Parameters
 525    ----------
 526    pulses : list[dict]
 527        All detected pulses within one apply->clear window.
 528    protocol_runs : list[dict]
 529        Parsed protocol runs (may be empty).
 530    apply_sample : int
 531        Sample index where the apply event fired.
 532    stimulus_index : int
 533        Index into the protocol's stimulus list for this trial.
 534
 535    Returns
 536    -------
 537    hyperpol : dict or None
 538        The hyperpolarization pulse, or None if not found.
 539    staircase_steps : list[dict]
 540        The remaining staircase steps.
 541    """
 542    if not pulses:
 543        return None, []
 544
 545    # Check if protocol says hyperpol should exist
 546    expects_hyperpol = False
 547    if protocol_runs:
 548        run = _find_protocol_run(apply_sample, protocol_runs)
 549        if run is not None:
 550            proto = run.get("protocol")
 551            if proto is not None and proto.hyperpolarization is not None:
 552                expects_hyperpol = True
 553
 554    first = pulses[0]
 555    if first["amplitude_pA"] < 0 and (expects_hyperpol or len(pulses) > 1):
 556        return first, pulses[1:]
 557
 558    return None, pulses

 561def compute_rmp(
 562    data: np.ndarray,
 563    display_scales: np.ndarray,
 564    baseline_start: int,
 565    baseline_end: int,
 566    vm_ch: int = VM_CH,
 567) -> float:
 568    """Compute resting membrane potential as the median baseline Vm.
 569
 570    Parameters
 571    ----------
 572    data : ndarray, shape (n_channels, n_samples)
 573        Raw recording.
 574    display_scales : ndarray
 575        Per-channel scale factors.
 576    baseline_start, baseline_end : int
 577        Sample indices defining the baseline window.
 578    vm_ch : int
 579        Channel index for membrane potential.
 580
 581    Returns
 582    -------
 583    float
 584        RMP in mV.
 585    """
 586    if baseline_end <= baseline_start:
 587        return float("nan")
 588    vm = data[vm_ch, baseline_start:baseline_end] * display_scales[vm_ch]
 589    return float(np.median(vm))

 592def compute_input_resistance(
 593    data: np.ndarray,
 594    display_scales: np.ndarray,
 595    hyperpol_pulse: dict,
 596    rmp_mV: float,
 597    sr: int,
 598    vm_ch: int = VM_CH,
 599) -> float:
 600    """Compute input resistance from the hyperpolarization pulse.
 601
 602    Uses the last 50% of the pulse to estimate steady-state Vm, then
 603    calculates Ri = delta_V / delta_I.
 604
 605    Parameters
 606    ----------
 607    data : ndarray, shape (n_channels, n_samples)
 608        Raw recording.
 609    display_scales : ndarray
 610        Per-channel scale factors.
 611    hyperpol_pulse : dict
 612        Must have ``onset``, ``offset``, ``amplitude_pA``.
 613    rmp_mV : float
 614        Resting membrane potential (mV) from baseline.
 615    sr : int
 616        Sampling rate in Hz.
 617    vm_ch : int
 618        Channel index for membrane potential.
 619
 620    Returns
 621    -------
 622    float
 623        Input resistance in MOhm, or NaN if calculation fails.
 624    """
 625    onset = hyperpol_pulse["onset"]
 626    offset = hyperpol_pulse["offset"]
 627    midpoint = onset + (offset - onset) // 2
 628
 629    if midpoint >= offset:
 630        return float("nan")
 631
 632    vm_during = data[vm_ch, midpoint:offset] * display_scales[vm_ch]
 633    steady_state_mV = float(np.median(vm_during))
 634
 635    delta_v_mV = steady_state_mV - rmp_mV
 636    delta_i_pA = hyperpol_pulse["amplitude_pA"]
 637
 638    if abs(delta_i_pA) < 1e-9:
 639        return float("nan")
 640
 641    # Ri = delta_V (mV) / delta_I (nA)  =>  mV / (pA / 1000) = MOhm
 642    ri_mohm = delta_v_mV / (delta_i_pA / 1000.0)
 643    return ri_mohm

 646def compute_all_intrinsics(
 647    data: np.ndarray,
 648    staircases: list[list[dict]],
 649    staircase_meta: list[dict],
 650    protocol_runs: list[dict],
 651    sr: int,
 652    display_scales: np.ndarray,
 653) -> list[dict]:
 654    """Compute RMP and Ri for every detected staircase.
 655
 656    For each staircase window, the baseline is the silent region between
 657    the apply event and the first detected pulse.  The hyperpolarization
 658    pulse (if present) is separated from the staircase steps and used to
 659    calculate input resistance.
 660
 661    Parameters
 662    ----------
 663    data : ndarray
 664        Full raw recording.
 665    staircases : list[list[dict]]
 666        Each staircase is a list of pulse dicts.
 667    staircase_meta : list[dict]
 668        Per-staircase metadata from detection.
 669    protocol_runs : list[dict]
 670        Parsed protocol runs.
 671    sr : int
 672        Sampling rate in Hz.
 673    display_scales : ndarray
 674        Per-channel scale factors.
 675
 676    Returns
 677    -------
 678    list[dict]
 679        One dict per staircase with keys: ``staircase_index``,
 680        ``stimulus_name``, ``t_start_s``, ``t_end_s``, ``n_steps``,
 681        ``min_amp_pA``, ``max_amp_pA``, ``rmp_mV``,
 682        ``input_resistance_MOhm``.
 683    """
 684    margin_samples = ms_to_samples(10, sr)
 685    results = []
 686
 687    for idx, (staircase, meta) in enumerate(zip(staircases, staircase_meta)):
 688        apply_sample = meta["apply_sample"]
 689        stim_index = meta.get("stimulus_index", -1)
 690
 691        # Separate hyperpol from staircase steps
 692        hyperpol, steps = separate_hyperpol(
 693            staircase, protocol_runs, apply_sample, stim_index
 694        )
 695
 696        # Baseline: from apply_sample to first pulse onset (minus margin)
 697        first_pulse_onset = staircase[0]["onset"]
 698        baseline_end = max(apply_sample, first_pulse_onset - margin_samples)
 699        rmp = compute_rmp(data, display_scales, apply_sample, baseline_end)
 700
 701        # Input resistance from hyperpol pulse
 702        if hyperpol is not None:
 703            ri = compute_input_resistance(
 704                data, display_scales, hyperpol, rmp, sr
 705            )
 706        else:
 707            ri = float("nan")
 708
 709        # Step amplitudes (from staircase steps only, not hyperpol)
 710        if steps:
 711            amps = [s["amplitude_pA"] for s in steps]
 712            t_start = steps[0]["onset"] / sr
 713            t_end = steps[-1]["offset"] / sr
 714        else:
 715            amps = [s["amplitude_pA"] for s in staircase]
 716            t_start = staircase[0]["onset"] / sr
 717            t_end = staircase[-1]["offset"] / sr
 718
 719        results.append({
 720            "staircase_index": idx,
 721            "stimulus_name": meta.get("stimulus_name", "unknown"),
 722            "t_start_s": round(t_start, 2),
 723            "t_end_s": round(t_end, 2),
 724            "n_steps": len(steps),
 725            "min_amp_pA": round(min(amps), 1) if amps else float("nan"),
 726            "max_amp_pA": round(max(amps), 1) if amps else float("nan"),
 727            "rmp_mV": round(rmp, 1),
 728            "input_resistance_MOhm": round(ri, 1) if not math.isnan(ri) else float("nan"),
 729        })
 730
 731    return results

 738def extract_step_segments(
 739    data: np.ndarray,
 740    steps: list[dict],
 741    sr: int,
 742    display_scales: np.ndarray,
 743    pad_ms: float = PAD_MS,
 744    vm_ch: int = VM_CH,
 745    cmd_ch: int = CMD_CH,
 746) -> list[dict]:
 747    """Extract aligned Vm and I_cmd segments for each step.
 748
 749    Each segment is aligned so that t=0 corresponds to step onset, with
 750    *pad_ms* of pre-step baseline shown at negative times.
 751
 752    Parameters
 753    ----------
 754    data : ndarray, shape (n_channels, n_samples)
 755        Full raw recording.
 756    steps : list[dict]
 757        Pulse dicts with ``onset``, ``offset``, ``amplitude_pA``.
 758    sr : int
 759        Sampling rate in Hz.
 760    display_scales : ndarray
 761        Per-channel scale factors.
 762    pad_ms : float
 763        Pre-onset baseline to include (ms).
 764    vm_ch, cmd_ch : int
 765        Channel indices.
 766
 767    Returns
 768    -------
 769    list[dict]
 770        Each dict has ``amplitude_pA``, ``time_ms``, ``vm_mV``,
 771        ``i_cmd_pA``.
 772    """
 773    pad_samples = ms_to_samples(pad_ms, sr)
 774    n_total = data.shape[1]
 775    segments = []
 776
 777    for step in steps:
 778        onset = step["onset"]
 779        offset = step["offset"]
 780
 781        seg_start = max(0, onset - pad_samples)
 782        seg_end = min(n_total, offset + pad_samples)
 783
 784        vm = data[vm_ch, seg_start:seg_end] * display_scales[vm_ch]
 785        i_cmd = data[cmd_ch, seg_start:seg_end] * display_scales[cmd_ch]
 786        t_ms = (
 787            (np.arange(seg_end - seg_start) - (onset - seg_start))
 788            / sr
 789            * 1000.0
 790        )
 791
 792        segments.append({
 793            "amplitude_pA": step["amplitude_pA"],
 794            "time_ms": t_ms,
 795            "vm_mV": vm,
 796            "i_cmd_pA": i_cmd,
 797        })
 798
 799    return segments

 806def add_scalebar(
 807    ax: plt.Axes,
 808    x_size: float,
 809    x_unit: str,
 810    y_size: float,
 811    y_unit: str,
 812    x_frac: float = 0.95,
 813    y_frac: float = -0.01,
 814) -> None:
 815    """Draw an L-shaped scale bar inside the axes.
 816
 817    The bar is anchored at a fractional (x, y) position within the axes
 818    and sized in data units.
 819
 820    Parameters
 821    ----------
 822    ax : Axes
 823        Target matplotlib axes.
 824    x_size, y_size : float
 825        Scale-bar lengths in data units.
 826    x_unit, y_unit : str
 827        Labels for each bar arm.
 828    x_frac, y_frac : float
 829        Fractional position of the bar's bottom-right corner
 830        (0 = left/bottom, 1 = right/top).
 831    """
 832    xlim = ax.get_xlim()
 833    ylim = ax.get_ylim()
 834    x_range = xlim[1] - xlim[0]
 835    y_range = ylim[1] - ylim[0]
 836
 837    x1 = xlim[0] + x_frac * x_range
 838    y0 = ylim[0] + y_frac * y_range
 839    x0 = x1 - x_size
 840    y1 = y0 + y_size
 841
 842    ax.plot([x0, x1], [y0, y0], color="black", linewidth=1.5, clip_on=False)
 843    ax.plot([x1, x1], [y0, y1], color="black", linewidth=1.5, clip_on=False)
 844    ax.text(
 845        (x0 + x1) / 2, y0 - 0.02 * y_range, f"{x_size} {x_unit}",
 846        ha="center", va="top", fontsize=9,
 847    )
 848    ax.text(
 849        x1 + 0.01 * x_range, (y0 + y1) / 2, f"{y_size} {y_unit}",
 850        ha="left", va="center", fontsize=9,
 851    )

 854def plot_step_overlay(
 855    segment_lists: list[list[dict]],
 856    title: str = "",
 857    figsize: tuple[float, float] = (6, 5),
 858    vm_scalebar: tuple = (100, "ms", 5, "mV"),
 859    cmd_scalebar: tuple = (100, "ms", 50, "pA"),
 860    vm_pad_mV: float = 2.0,
 861    cmd_pad_pA: float = 5.0,
 862    trace_alpha: float = 0.85,
 863    sweep_labels: list | None = None,
 864) -> tuple[plt.Figure, tuple[plt.Axes, plt.Axes]]:
 865    """Plot overlaid step responses.
 866
 867    Creates a two-panel figure: membrane potential on top (3:1 height
 868    ratio) and command current on the bottom, with minimal aesthetics
 869    (no spines/ticks), L-shaped scale bars, and an RMP marker.
 870
 871    Coloring:
 872        - If ``sweep_labels`` is None, traces are colored by injection
 873          amplitude (black→blue CMAP).
 874        - If ``sweep_labels`` is provided (one label per sweep), each
 875          sweep gets a single distinct color sampled from ``Spectral``
 876          (all traces in that sweep share it). A legend is drawn.
 877
 878    Parameters
 879    ----------
 880    segment_lists : list[list[dict]]
 881        One inner list per sweep; each entry is a segment dict from
 882        :func:`extract_step_segments`.
 883    title : str
 884        Optional figure title.
 885    figsize : tuple
 886        Figure dimensions (width, height) in inches.
 887    vm_scalebar, cmd_scalebar : tuple
 888        (x_size, x_unit, y_size, y_unit) for each panel's scale bar.
 889    vm_pad_mV, cmd_pad_pA : float
 890        Y-axis padding around data extremes.
 891    trace_alpha : float
 892        Opacity of each trace.
 893    sweep_labels : list or None
 894        Optional labels, one per sweep. If given, switches coloring
 895        from amplitude-based to sweep-index-based and draws a legend.
 896
 897    Returns
 898    -------
 899    fig : Figure
 900    axes : (ax_vm, ax_cmd)
 901    """
 902    color_by_sweep = sweep_labels is not None
 903
 904    if color_by_sweep:
 905        n_sweeps = len(segment_lists)
 906        spectral = plt.get_cmap("Spectral")
 907        sweep_base_colors = [
 908            spectral(i / max(1, n_sweeps - 1)) for i in range(n_sweeps)
 909        ]
 910    else:
 911        all_amps = sorted({
 912            seg["amplitude_pA"]
 913            for sweep_segs in segment_lists
 914            for seg in sweep_segs
 915        })
 916        amp_norm = mcolors.Normalize(vmin=min(all_amps), vmax=max(all_amps))
 917
 918    fig, (ax_vm, ax_cmd) = plt.subplots(
 919        2, 1, figsize=figsize, sharex=True,
 920        gridspec_kw={"height_ratios": [3, 1], "hspace": 0.2},
 921    )
 922
 923    for sweep_idx, sweep_segs in enumerate(segment_lists):
 924        for seg in sweep_segs:
 925            if color_by_sweep:
 926                color = sweep_base_colors[sweep_idx]
 927            else:
 928                color = CMAP(amp_norm(seg["amplitude_pA"]))
 929            ax_vm.plot(
 930                seg["time_ms"], seg["vm_mV"],
 931                color=color, linewidth=0.9, alpha=trace_alpha,
 932            )
 933            ax_cmd.plot(
 934                seg["time_ms"], seg["i_cmd_pA"],
 935                color=color, linewidth=0.9, alpha=trace_alpha,
 936            )
 937
 938    # Tight y-limits from data
 939    vm_vals = np.concatenate([
 940        seg["vm_mV"] for segs in segment_lists for seg in segs
 941    ])
 942    cmd_vals = np.concatenate([
 943        seg["i_cmd_pA"] for segs in segment_lists for seg in segs
 944    ])
 945    ax_vm.set_ylim(vm_vals.min() - vm_pad_mV, vm_vals.max() + vm_pad_mV)
 946    ax_cmd.set_ylim(cmd_vals.min() - cmd_pad_pA, cmd_vals.max() + cmd_pad_pA)
 947
 948    # Remove spines, ticks, and labels
 949    for ax in (ax_vm, ax_cmd):
 950        for spine in ax.spines.values():
 951            spine.set_visible(False)
 952        ax.set_xticks([])
 953        ax.set_yticks([])
 954        ax.set_xlabel("")
 955        ax.set_ylabel("")
 956
 957    # RMP marker from pre-step baseline
 958    baseline_vals = []
 959    for sweep_segs in segment_lists:
 960        for seg in sweep_segs:
 961            mask = seg["time_ms"] < 0
 962            if np.any(mask):
 963                baseline_vals.append(np.median(seg["vm_mV"][mask]))
 964    rmp = np.median(baseline_vals)
 965    rmp_rounded = round(rmp)
 966
 967    xlim = ax_vm.get_xlim()
 968    tick_x = xlim[0]
 969    ax_vm.plot(tick_x, rmp, marker="_", markersize=8, color="black",
 970               clip_on=False)
 971    ax_vm.text(
 972        tick_x - 0.02 * (xlim[1] - xlim[0]), rmp, f"{rmp_rounded} mV",
 973        ha="right", va="center", fontsize=9,
 974    )
 975
 976    add_scalebar(ax_vm, *vm_scalebar)
 977    add_scalebar(ax_cmd, *cmd_scalebar)
 978
 979    if color_by_sweep:
 980        legend_handles = [
 981            plt.Line2D([0], [0], color=sweep_base_colors[i], linewidth=1.5,
 982                       label=str(sweep_labels[i]))
 983            for i in range(len(segment_lists))
 984        ]
 985        ax_vm.legend(
 986            handles=legend_handles, loc="upper right",
 987            fontsize=8, frameon=False, title="Trial",
 988            title_fontsize=8,
 989        )
 990
 991    if title:
 992        fig.suptitle(title, fontsize=13, fontweight="bold")
 993
 994    fig.tight_layout()
 995    return fig, (ax_vm, ax_cmd)

1117def parse_selection(user_input: str, n: int) -> list[int] | None:
1118    """Parse a user selection string into a list of indices.
1119
1120    Accepted formats: ``"3"``, ``"0-5"``, ``"0,3,7"``, ``"all"``.
1121
1122    Parameters
1123    ----------
1124    user_input : str
1125        Raw input string.
1126    n : int
1127        Total number of available staircases.
1128
1129    Returns
1130    -------
1131    list[int] or None
1132        Sorted list of valid indices, or None if input is invalid.
1133    """
1134    text = user_input.strip().lower()
1135    if text == "all":
1136        return list(range(n))
1137
1138    indices = set()
1139    for part in text.split(","):
1140        part = part.strip()
1141        if "-" in part:
1142            bounds = part.split("-", 1)
1143            try:
1144                lo, hi = int(bounds[0]), int(bounds[1])
1145            except ValueError:
1146                return None
1147            indices.update(range(lo, hi + 1))
1148        else:
1149            try:
1150                indices.add(int(part))
1151            except ValueError:
1152                return None
1153
1154    valid = sorted(i for i in indices if 0 <= i < n)
1155    return valid if valid else None

1217def interactive_loop(
1218    h5_path: Path,
1219    data: np.ndarray,
1220    staircases: list[list[dict]],
1221    staircase_meta: list[dict],
1222    protocol_runs: list[dict],
1223    intrinsics: list[dict],
1224    sr: int,
1225    display_scales: np.ndarray,
1226) -> None:
1227    """Run the interactive plot selection loop.
1228
1229    Prompts the user to select staircases to plot, shows the overlay,
1230    and asks whether to save each figure.
1231
1232    Parameters
1233    ----------
1234    h5_path : Path
1235        Source HDF5 file (used for figure naming).
1236    data : ndarray
1237        Full raw recording.
1238    staircases, staircase_meta : list
1239        Detected staircases and metadata.
1240    protocol_runs : list[dict]
1241        Parsed protocol runs.
1242    intrinsics : list[dict]
1243        Pre-computed intrinsic properties.
1244    sr : int
1245        Sampling rate.
1246    display_scales : ndarray
1247        Per-channel scale factors.
1248    """
1249    n = len(staircases)
1250    overlay_mode = True  # toggled with "m"; controls behavior for multi-select
1251
1252    while True:
1253        mode_tag = "overlay" if overlay_mode else "separate"
1254        choice = input(
1255            f'[{mode_tag}] Enter staircase number(s) to plot '
1256            f'(e.g. "0", "0-5", "all"), "m" to toggle mode, or "q" to quit: '
1257        ).strip()
1258
1259        if choice.lower() == "q":
1260            break
1261        if choice.lower() in ("m", "mode"):
1262            overlay_mode = not overlay_mode
1263            new_tag = "overlay" if overlay_mode else "separate"
1264            print(f"  Multi-select mode: {new_tag}\n")
1265            continue
1266
1267        indices = parse_selection(choice, n)
1268        if indices is None:
1269            print("Invalid selection. Try again.\n")
1270            continue
1271
1272        # Single index: plot alone, colored by amplitude.
1273        # Multi-index in overlay mode: one figure colored by staircase index.
1274        # Multi-index in separate mode: one figure per staircase (legacy).
1275        if len(indices) == 1 or not overlay_mode:
1276            for idx in indices:
1277                staircase = staircases[idx]
1278                meta = staircase_meta[idx]
1279                _, steps = separate_hyperpol(
1280                    staircase, protocol_runs,
1281                    meta["apply_sample"], meta.get("stimulus_index", -1),
1282                )
1283
1284                if not steps:
1285                    print(f"  Staircase {idx}: no steps to plot.")
1286                    continue
1287
1288                segments = extract_step_segments(data, steps, sr, display_scales)
1289                fig, _ = plot_step_overlay([segments], title=None)
1290                plt.show(block=False)
1291                plt.pause(0.1)
1292
1293                save_choice = input(
1294                    f"  Save staircase {idx} figure? [y/n]: "
1295                ).strip().lower()
1296                if save_choice == "y":
1297                    fig_name = f"{h5_path.stem}_staircase{idx}_overlay"
1298                    save_fig_both(fig, fig_name, overwrite=True)
1299
1300                plt.close(fig)
1301        else:
1302            segment_lists: list[list[dict]] = []
1303            used_indices: list[int] = []
1304            for idx in indices:
1305                staircase = staircases[idx]
1306                meta = staircase_meta[idx]
1307                _, steps = separate_hyperpol(
1308                    staircase, protocol_runs,
1309                    meta["apply_sample"], meta.get("stimulus_index", -1),
1310                )
1311                if not steps:
1312                    print(f"  Staircase {idx}: no steps, skipping.")
1313                    continue
1314                segment_lists.append(
1315                    extract_step_segments(data, steps, sr, display_scales)
1316                )
1317                used_indices.append(idx)

243def find_pulses(cmd_pA: np.ndarray, sr: int,
244                min_amp_pA: float = MIN_AMP_PA,
245                min_step_ms: float = MIN_STEP_MS) -> list[dict]:
246    """Find contiguous current-injection pulses above a threshold.
247
248    Detects every run of samples where ``|cmd| >= min_amp_pA`` that lasts
249    at least ``min_step_ms``.  The amplitude of each pulse is the median
250    of the smoothed command signal during the run.
251
252    Parameters
253    ----------
254    cmd_pA : ndarray
255        Smoothed command-current waveform in pA (1-D).
256    sr : int
257        Sampling rate in Hz.
258    min_amp_pA : float
259        Minimum absolute amplitude to consider (pA).
260    min_step_ms : float
261        Minimum duration for a valid pulse (ms).
262
263    Returns
264    -------
265    list[dict]
266        Each dict has ``onset`` (int), ``offset`` (int), and
267        ``amplitude_pA`` (float).  Indices are relative to *cmd_pA*.
268    """
269    min_samples = ms_to_samples(min_step_ms, sr)
270    above = np.abs(cmd_pA) >= min_amp_pA
271
272    ab = above.astype(np.int8)
273    edges = np.diff(ab)
274    rising = np.flatnonzero(edges == 1) + 1
275    falling = np.flatnonzero(edges == -1) + 1
276    if above[0]:
277        rising = np.concatenate(([0], rising))
278    if above[-1]:
279        falling = np.concatenate((falling, [len(above)]))
280
281    pulses = []
282    for onset, offset in zip(rising, falling):
283        if offset - onset < min_samples:
284            continue
285        amp = float(np.median(cmd_pA[onset:offset]))
286        pulses.append({
287            "onset": int(onset),
288            "offset": int(offset),
289            "amplitude_pA": amp,
290        })
291    return pulses

596    hyperpol_pulse: dict,
597    rmp_mV: float,
598    sr: int,
599    vm_ch: int = VM_CH,
600) -> float:
601    """Compute input resistance from the hyperpolarization pulse.
602
603    Uses the last 50% of the pulse to estimate steady-state Vm, then
604    calculates Ri = delta_V / delta_I.
605
606    Parameters
607    ----------
608    data : ndarray, shape (n_channels, n_samples)
609        Raw recording.
610    display_scales : ndarray
611        Per-channel scale factors.
612    hyperpol_pulse : dict
613        Must have ``onset``, ``offset``, ``amplitude_pA``.
614    rmp_mV : float
615        Resting membrane potential (mV) from baseline.
616    sr : int
617        Sampling rate in Hz.
618    vm_ch : int
619        Channel index for membrane potential.
620
621    Returns
622    -------
623    float
624        Input resistance in MOhm, or NaN if calculation fails.
625    """
626    onset = hyperpol_pulse["onset"]
627    offset = hyperpol_pulse["offset"]

629
630    if midpoint >= offset:
631        return float("nan")
632
633    vm_during = data[vm_ch, midpoint:offset] * display_scales[vm_ch]
634    steady_state_mV = float(np.median(vm_during))

638

640        return float("nan")
641
642    # Ri = delta_V (mV) / delta_I (nA)  =>  mV / (pA / 1000) = MOhm
643    ri_mohm = delta_v_mV / (delta_i_pA / 1000.0)
644    return ri_mohm
