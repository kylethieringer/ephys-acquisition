ui.main_window

MainWindow — top-level Qt window that assembles all panels.

Layout::

QMainWindow
└── central widget (vertical)
    ├── TopChromeBar  (session label + mode/clamp pills + status badge)
    ├── QHBoxLayout
    │   ├── Sidebar (icon rail: Acquire / Protocol / Channels / Setup)
    │   └── QVBoxLayout
    │       ├── QStackedWidget
    │       │   ├── AcquirePage  (traces + camera + subject/protocol/stim)
    │       │   ├── ProtocolPage (saved-list + inline ProtocolBuilderPanel)
    │       │   ├── ChannelsPage (per-channel table)
    │       │   └── SetupPage    (2x2 card grid: DAQ / Mapping / Save / Camera)
    │       └── Recording bar (always visible, bottom)

Developer notes

Both acquisition controllers are always instantiated; only the active one is started when the user clicks Start. The ring buffer from ~acquisition.continuous_mode.ContinuousAcquisition is shared by both modes.

The per-channel QCheckBox + ChannelYControls widgets live on the Channels page (only page that owns them). _on_clamp_mode_changed mutates those widgets in place so switching pages does not lose state.

774    def _on_start_protocol(self) -> None:
775        if self._pending_protocol is None:
776            return
777        if not self._check_metadata():
778            return
779        try:
780            protocol_dict = dict(self._pending_protocol)
781            save_dir = protocol_dict.pop("save_dir", self._ctrl_panel.save_dir)
782            protocol = protocol_from_dict(protocol_dict)
783            metadata = self._ctrl_panel.get_metadata()
784
785            self._apply_channel_defs(protocol.clamp_mode)
786
787            if self._active_mode == "continuous":
788                if not self._acq.is_running:
789                    self._acq.start()
790                self._acq.start_recording(save_dir, metadata)
791                self._acq.start_protocol(protocol)
792                n = len(protocol.stimuli) * protocol.repeats_per_stimulus
793                self._ctrl_panel.set_status(
794                    f"Protocol '{protocol.name}' running (continuous) — {n} trials…"
795                )
796            else:
797                if not self._trial_acq.is_running:
798                    self._trial_acq.start()
799                self._trial_acq.run_protocol(protocol, save_dir, metadata)
800                n = len(protocol.stimuli) * protocol.repeats_per_stimulus
801                self._ctrl_panel.set_status(
802                    f"Protocol '{protocol.name}' running — {n} trials total…"
803                )
804            self._ctrl_panel.enable_run_protocol_button(False)
805            self._ctrl_panel.enable_stop_protocol_button(True)
806            self._chrome.status_badge.set_state("protocol")
807        except Exception as exc:

775    def _on_start_protocol(self) -> None:
776        if self._pending_protocol is None:
777            return
778        if not self._check_metadata():
779            return
780        try:
781            protocol_dict = dict(self._pending_protocol)
782            save_dir = protocol_dict.pop("save_dir", self._ctrl_panel.save_dir)
783            protocol = protocol_from_dict(protocol_dict)
784            metadata = self._ctrl_panel.get_metadata()
785
786            self._apply_channel_defs(protocol.clamp_mode)
787
788            if self._active_mode == "continuous":
789                if not self._acq.is_running:
790                    self._acq.start()
791                self._acq.start_recording(save_dir, metadata)
792                self._acq.start_protocol(protocol)
793                n = len(protocol.stimuli) * protocol.repeats_per_stimulus
794                self._ctrl_panel.set_status(
795                    f"Protocol '{protocol.name}' running (continuous) — {n} trials…"
796                )
797            else:
798                if not self._trial_acq.is_running:
799                    self._trial_acq.start()
800                self._trial_acq.run_protocol(protocol, save_dir, metadata)
801                n = len(protocol.stimuli) * protocol.repeats_per_stimulus
802                self._ctrl_panel.set_status(
803                    f"Protocol '{protocol.name}' running — {n} trials total…"
804                )
805            self._ctrl_panel.enable_run_protocol_button(False)
806            self._ctrl_panel.enable_stop_protocol_button(True)
807            self._chrome.status_badge.set_state("protocol")
808        except Exception as exc:
809            self._on_error(str(exc))
