analysis.align_video

Align video frames with ephys data using the TTL trigger channel.

Produces a composite video where each frame shows:

  • Top panel: camera image at time t
  • Bottom panel: membrane potential (mV) trace, ±1 s centered on t, with a centre line marking the current frame time

Frame timestamps are derived from rising edges on channel 4 (TTLLoopback), which fires one pulse per camera frame.

Usage — continuous-mode HDF5: python align_video.py recording.h5 video.avi output.avi

Usage — trial-mode HDF5 (pick one trial): python align_video.py trials.h5 video.avi output.avi --trial 3

If output path is omitted, the file is written next to the HDF5 as _aligned.avi (continuous) or _trial_aligned.avi (trial).

 55_AX  = (120, 120, 130)   # axis / tick grey
 56_TXT = (220, 220, 230)   # label text

 64def _is_trial_mode(h5_path: Path) -> bool:
 65    with h5py.File(h5_path, "r") as f:
 66        return any(k.startswith("trial_") for k in f.keys())

 69def _load_continuous(h5_path: Path) -> tuple[np.ndarray, np.ndarray, float]:
 70    """Return (vm_mV, ttl_V, sample_rate) from a continuous-mode HDF5."""
 71    with h5py.File(h5_path, "r") as f:
 72        sr     = float(f["metadata"].attrs["sample_rate"])
 73        scales = f["metadata"]["display_scales"][:]
 74        data   = f["data"]["analog_input"][:]
 75    vm  = data[VM_CHANNEL]  * scales[VM_CHANNEL]
 76    ttl = data[TTL_CHANNEL] * scales[TTL_CHANNEL]
 77    return vm, ttl, sr

 79
 80def _load_trial(h5_path: Path, trial_num: int) -> tuple[np.ndarray, np.ndarray, float]:
 81    """Return (vm_mV, ttl_V, sample_rate) from trial N of a trial-mode HDF5."""
 82    key = f"trial_{trial_num:03d}"
 83    with h5py.File(h5_path, "r") as f:
 84        available = sorted(k for k in f.keys() if k.startswith("trial_"))
 85        if key not in f:
 86            raise KeyError(f"'{key}' not found in {h5_path.name}. Available: {available}")
 87        sr     = float(f["metadata"].attrs["sample_rate"])
 88        scales = f["metadata"]["display_scales"][:]
 89        data   = f[key]["analog_input"][:]
 90    vm  = data[VM_CHANNEL]  * scales[VM_CHANNEL]
 91    ttl = data[TTL_CHANNEL] * scales[TTL_CHANNEL]
 92    return vm, ttl, sr

 99def find_frame_samples(ttl: np.ndarray, threshold: float = TTL_THRESHOLD) -> np.ndarray:
100    """Return sample indices of rising edges in the TTL trigger signal.
101
102    Each rising edge marks the moment the DAQ fired a trigger pulse to
103    the camera, so edge i corresponds to the start of video frame i.
104    """
105    above = (ttl >= threshold).astype(np.int8)
106    diff  = np.diff(above, prepend=np.int8(0))
107    return np.where(diff == 1)[0]

114def _y_range(vm: np.ndarray, pad_frac: float = 0.08) -> tuple[float, float]:
115    """Y-axis range using the full min/max of the Vm trace."""
116    lo = float(np.nanmin(vm))
117    hi = float(np.nanmax(vm))
118    margin = (hi - lo) * pad_frac
119    return lo - margin, hi + margin

122def _compute_rmp(vm: np.ndarray) -> float:
123    """Estimate resting membrane potential as the median of the Vm trace.
124
125    The median is robust to action potentials, which are brief positive
126    deflections that do not strongly affect the middle of the distribution.
127    """
128    return float(np.nanmedian(vm))

131def _minmax_envelope(window: np.ndarray, n_out: int) -> tuple[np.ndarray, np.ndarray]:
132    """Downsample *window* to *n_out* columns using a min-max envelope.
133
134    For each output pixel column, the minimum and maximum of all input
135    samples that fall in that column's time bin are returned.  This
136    preserves the full amplitude of transients (spikes) regardless of
137    how many raw samples map to each pixel.
138
139    Returns:
140        col_min, col_max — each shape (n_out,), NaN where no valid samples.
141    """
142    n     = len(window)
143    edges = np.linspace(0, n, n_out + 1)
144    lo_idx = np.floor(edges[:-1]).astype(int)
145    hi_idx = np.minimum(np.ceil(edges[1:]).astype(int), n)
146
147    col_min = np.full(n_out, np.nan)
148    col_max = np.full(n_out, np.nan)
149    for col in range(n_out):
150        seg   = window[lo_idx[col] : hi_idx[col]]
151        valid = seg[~np.isnan(seg)]
152        if len(valid):
153            col_min[col] = valid.min()
154            col_max[col] = valid.max()
155    return col_min, col_max

158def _draw_trace(
159    vm:             np.ndarray,
160    center_sample:  int,
161    sr:             float,
162    width:          int,
163    y_lo:           float,
164    y_hi:           float,
165    rmp:            float,
166    height:         int = TRACE_HEIGHT,
167) -> np.ndarray:
168    """Render a BGR trace image (height × width) centred on *center_sample*.
169
170    The x axis spans [center_sample - HALF_WIN_S*sr, center_sample + HALF_WIN_S*sr].
171    A dashed vertical line marks the current time (centre, t = 0).
172    A faint horizontal line marks the resting membrane potential.
173    Regions outside the recording are left as background.
174    The trace is rendered using a min-max envelope so spike amplitudes are
175    preserved regardless of how many raw samples map to each display pixel.
176    """
177    half  = int(HALF_WIN_S * sr)
178    n     = len(vm)
179    lo    = center_sample - half
180    hi    = center_sample + half
181
182    # Extract window, padding with NaN where outside the recording
183    window = np.full(hi - lo, np.nan)
184    src_lo = max(lo, 0)
185    src_hi = min(hi, n)
186    if src_lo < src_hi:
187        window[src_lo - lo : src_hi - lo] = vm[src_lo:src_hi]
188
189    # ---- layout constants ----
190    pad_top = 18
191    pad_bot = 32
192    pad_lft = 48   # room for y-axis labels
193    draw_h  = height - pad_top - pad_bot
194    draw_w  = width  - pad_lft
195    baseline = pad_top + draw_h
196
197    img = np.full((height, width, 3), _BG, dtype=np.uint8)
198
199    def _y_px(v: float) -> int:
200        frac = (v - y_lo) / (y_hi - y_lo + 1e-12)
201        return int(pad_top + (1.0 - frac) * draw_h)
202
203    def _put(text: str, x: int, y: int, scale: float = 0.38,
204             color: tuple = _TXT) -> None:
205        cv2.putText(img, text, (x, y),
206                    cv2.FONT_HERSHEY_SIMPLEX, scale, color, 1, cv2.LINE_AA)
207
208    # ---- y-axis: only limits + RMP ----
209    for v, color in [(y_lo, _AX), (rmp, _RMP), (y_hi, _AX)]:
210        py = _y_px(v)
211        cv2.line(img, (pad_lft - 6, py), (pad_lft, py), color, 1)
212        label = f"{v:.0f}"
213        (tw, th), _ = cv2.getTextSize(label, cv2.FONT_HERSHEY_SIMPLEX, 0.35, 1)
214        _put(label, pad_lft - 7 - tw, py + th // 2, scale=0.35, color=color)
215
216    _put("mV", 2, pad_top, scale=0.35)
217
218    # faint dashed horizontal line at RMP across the plot area
219    rmp_py = _y_px(rmp)
220    for dx in range(pad_lft, width, 8):
221        cv2.line(img, (dx, rmp_py), (min(dx + 4, width - 1), rmp_py), _RMP, 1)
222
223    # left axis line
224    cv2.line(img, (pad_lft, pad_top), (pad_lft, baseline), _AX, 1)
225    # bottom axis line
226    cv2.line(img, (pad_lft, baseline), (width - 1, baseline), _AX, 1)
227
228    # ---- x-axis ticks ----
229    for t in (-1.0, -0.5, 0.0, 0.5, 1.0):
230        frac = (t + HALF_WIN_S) / (2 * HALF_WIN_S)
231        px   = pad_lft + int(frac * (draw_w - 1))
232        cv2.line(img, (px, baseline), (px, baseline + 4), _AX, 1)
233        label = f"{t:+.1f}s" if t != 0.0 else "0"
234        (tw, _), _ = cv2.getTextSize(label, cv2.FONT_HERSHEY_SIMPLEX, 0.35, 1)
235        _put(label, px - tw // 2, height - 4, scale=0.35)
236
237    # ---- centre line (dashed) ----
238    cx = pad_lft + draw_w // 2
239    for dy in range(pad_top, baseline, 6):
240        cv2.line(img, (cx, dy), (cx, min(dy + 3, baseline)), _CL, 1)
241
242    # ---- trace (min-max envelope as polylines) ----
243    col_min, col_max = _minmax_envelope(window, draw_w)
244    seg_max: list[tuple[int, int]] = []
245    seg_min: list[tuple[int, int]] = []
246    segs_max: list[list[tuple[int, int]]] = []
247    segs_min: list[list[tuple[int, int]]] = []
248    for col in range(draw_w):
249        if np.isnan(col_min[col]):
250            if seg_max:
251                segs_max.append(seg_max); segs_min.append(seg_min)
252                seg_max = []; seg_min = []
253            continue
254        x = pad_lft + col
255        seg_max.append((x, _y_px(float(col_max[col]))))
256        seg_min.append((x, _y_px(float(col_min[col]))))
257    if seg_max:
258        segs_max.append(seg_max); segs_min.append(seg_min)
259
260    for seg in segs_max:
261        cv2.polylines(img, [np.array(seg, dtype=np.int32)], False, _FG, 1, cv2.LINE_AA)
262    for seg in segs_min:
263        cv2.polylines(img, [np.array(seg, dtype=np.int32)], False, _FG, 1, cv2.LINE_AA)
264
265    return img

272def align(
273    h5_path:    Path,
274    video_path: Path,
275    output_path: Path,
276    trial_num:  int | None = None,
277) -> None:
278    """Create the aligned composite video.
279
280    Args:
281        h5_path:     Path to the HDF5 electrophysiology file.
282        video_path:  Path to the input camera video (.avi).
283        output_path: Destination for the composite video.
284        trial_num:   Trial number (1-based) for trial-mode HDF5.
285                     Ignored for continuous-mode files.
286    """
287    # --- load ephys --------------------------------------------------------
288    if _is_trial_mode(h5_path):
289        if trial_num is None:
290            trial_num = 1
291            print(f"[info] Trial-mode HDF5 detected; defaulting to trial {trial_num}.")
292        print(f"[info] Loading trial {trial_num} from {h5_path.name} …")
293        vm, ttl, sr = _load_trial(h5_path, trial_num)
294    else:
295        print(f"[info] Loading continuous recording from {h5_path.name} …")
296        vm, ttl, sr = _load_continuous(h5_path)
297
298    # --- find frame timestamps ---------------------------------------------
299    frame_samples = find_frame_samples(ttl)
300    n_edges = len(frame_samples)
301    if n_edges == 0:
302        sys.exit(
303            "[error] No TTL rising edges found in channel 4 (TTLLoopback). "
304            "Check that the correct HDF5 file and TTL channel are used."
305        )
306    print(f"[info] Found {n_edges} TTL rising edges → frame timestamps.")
307
308    # --- open video --------------------------------------------------------
309    cap = cv2.VideoCapture(str(video_path))
310    if not cap.isOpened():
311        sys.exit(f"[error] Cannot open video: {video_path}")
312
313    n_video_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
314    fps            = cap.get(cv2.CAP_PROP_FPS)
315    vid_w          = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
316    vid_h          = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
317
318    n_frames = min(n_edges, n_video_frames)
319    if n_edges != n_video_frames:
320        print(
321            f"[warn] TTL edges ({n_edges}) ≠ video frames ({n_video_frames}). "
322            f"Using first {n_frames} frames."
323        )
324
325    # --- prepare output writer --------------------------------------------
326    out_h   = vid_h + TRACE_HEIGHT
327    fourcc  = cv2.VideoWriter_fourcc(*"MJPG")
328    writer  = cv2.VideoWriter(str(output_path), fourcc, fps, (vid_w, out_h))
329    if not writer.isOpened():
330        sys.exit(f"[error] Cannot open output video for writing: {output_path}")
331
332    # --- pre-compute y range and RMP from the full Vm trace ---------------
333    y_lo, y_hi = _y_range(vm)
334    rmp = _compute_rmp(vm)
335    print(f"[info] Resting membrane potential: {rmp:.1f} mV")
336
337    # --- render frames -----------------------------------------------------
338    print(f"[info] Writing {n_frames} frames to {output_path.name} …")
339    for i in range(n_frames):
340        ret, frame = cap.read()
341        if not ret:
342            print(f"[warn] Could not read video frame {i}; stopping early.")
343            break
344
345        center_sample = int(frame_samples[i])
346        trace_img     = _draw_trace(vm, center_sample, sr, vid_w, y_lo, y_hi, rmp)
347
348        # Resize frame to match video width if needed (should be a no-op)
349        if frame.shape[1] != vid_w:
350            frame = cv2.resize(frame, (vid_w, vid_h))
351
352        composite = np.vstack([frame, trace_img])
353        writer.write(composite)
354
355        # Progress: print every 5% of frames
356        step = max(1, n_frames // 20)
357        if i % step == 0 or i == n_frames - 1:
358            pct = 100 * (i + 1) / n_frames
359            print(f"\r  {pct:5.1f}%  frame {i+1}/{n_frames}", end="", flush=True)
360
361    print()  # newline after progress
362
363    cap.release()
364    writer.release()
365    print(f"[done] Saved → {output_path}")

372def _gui_select_files() -> tuple[Path, Path, Path, int | None] | None:
373    """Show a single window where the user can review and change file selections.
374
375    Returns (h5_path, video_path, output_path, trial_num) or None if cancelled.
376    """
377    result: list = [None]  # mutable container so nested functions can write it
378
379    root = tk.Tk()
380    root.title("Align Video — File Selection")
381    root.resizable(False, False)
382
383    pad = dict(padx=8, pady=4)
384
385    # --- row helpers ----------------------------------------------------------
386    def _make_row(parent, row: int, label: str, browse_cmd):
387        tk.Label(parent, text=label, anchor="w").grid(
388            row=row, column=0, sticky="w", **pad,
389        )
390        var = tk.StringVar()
391        entry = tk.Entry(parent, textvariable=var, width=60)
392        entry.grid(row=row, column=1, sticky="ew", **pad)
393        tk.Button(parent, text="Browse…", command=browse_cmd).grid(
394            row=row, column=2, **pad,
395        )
396        return var, entry
397
398    # --- browse callbacks -----------------------------------------------------
399    def _browse_h5():
400        path = filedialog.askopenfilename(
401            title="Select HDF5 data file",
402            filetypes=[("HDF5 files", "*.h5 *.hdf5"), ("All files", "*.*")],
403        )
404        if path:
405            h5_var.set(path)
406            _update_trial_row()
407            # auto-fill output name based on new h5 selection
408            if not out_var.get():
409                _suggest_output()
410
411    def _browse_video():
412        path = filedialog.askopenfilename(
413            title="Select video file",
414            filetypes=[("Video files", "*.avi *.mp4"), ("All files", "*.*")],
415        )
416        if path:
417            video_var.set(path)
418
419    def _browse_output():
420        init_dir = ""
421        h5_str = h5_var.get()
422        if h5_str:
423            init_dir = str(Path(h5_str).parent)
424        path = filedialog.asksaveasfilename(
425            title="Save aligned video as",
426            defaultextension=".avi",
427            filetypes=[("AVI video", "*.avi"), ("All files", "*.*")],
428            initialdir=init_dir,
429        )
430        if path:
431            out_var.set(path)
432
433    def _suggest_output():
434        h5_str = h5_var.get()
435        if h5_str:
436            p = Path(h5_str)
437            out_var.set(str(p.parent / f"{p.stem}_aligned.avi"))
438
439    # --- trial row visibility -------------------------------------------------
440    trial_var = tk.StringVar(value="1")
441    trial_widgets: list[tk.Widget] = []
442
443    def _update_trial_row():
444        h5_str = h5_var.get()
445        is_trial = False
446        if h5_str and Path(h5_str).exists():
447            try:
448                is_trial = _is_trial_mode(Path(h5_str))
449            except Exception:
450                pass
451        for w in trial_widgets:
452            if is_trial:
453                w.grid()
454            else:
455                w.grid_remove()
456
457    # --- build rows -----------------------------------------------------------
458    h5_var, _    = _make_row(root, 0, "Data file (.h5):", _browse_h5)
459    video_var, _ = _make_row(root, 1, "Video file:",      _browse_video)
460    out_var, _   = _make_row(root, 2, "Output file:",     _browse_output)
461
462    # trial number row (hidden by default)
463    lbl = tk.Label(root, text="Trial number:", anchor="w")
464    lbl.grid(row=3, column=0, sticky="w", **pad)
465    trial_entry = tk.Entry(root, textvariable=trial_var, width=10)
466    trial_entry.grid(row=3, column=1, sticky="w", **pad)
467    trial_widgets.extend([lbl, trial_entry])
468    for w in trial_widgets:
469        w.grid_remove()
470
471    root.columnconfigure(1, weight=1)
472
473    # --- buttons --------------------------------------------------------------
474    btn_frame = tk.Frame(root)
475    btn_frame.grid(row=4, column=0, columnspan=3, pady=10)
476
477    def _on_run():
478        h5_str    = h5_var.get().strip()
479        vid_str   = video_var.get().strip()
480        out_str   = out_var.get().strip()
481        if not h5_str or not vid_str or not out_str:
482            messagebox.showwarning("Missing fields", "Please fill in all three file paths.")
483            return
484        trial_num = None
485        h5_path = Path(h5_str).resolve()
486        if h5_path.exists():
487            try:
488                if _is_trial_mode(h5_path):
489                    try:
490                        trial_num = int(trial_var.get())
491                    except ValueError:
492                        messagebox.showwarning("Invalid trial", "Trial number must be an integer.")
493                        return
494            except Exception:
495                pass
496        result[0] = (h5_path, Path(vid_str).resolve(), Path(out_str).resolve(), trial_num)
497        root.destroy()
498
499    def _on_cancel():
500        root.destroy()
501
502    tk.Button(btn_frame, text="Run", width=10, command=_on_run).pack(side="left", padx=8)
503    tk.Button(btn_frame, text="Cancel", width=10, command=_on_cancel).pack(side="left", padx=8)
504
505    root.protocol("WM_DELETE_WINDOW", _on_cancel)
506    root.mainloop()
507
508    return result[0]

522def main() -> None:
523    global VM_CHANNEL, TTL_CHANNEL, HALF_WIN_S, TRACE_HEIGHT
524
525    # No CLI args → launch GUI file picker
526    if len(sys.argv) == 1:
527        if not _HAS_TK:
528            sys.exit("[error] tkinter is not available. Provide file paths as CLI arguments.")
529        result = _gui_select_files()
530        if result is None:
531            sys.exit("[info] Cancelled.")
532        h5_path, video_path, out_path, trial_num = result
533        if not h5_path.exists():
534            sys.exit(f"[error] HDF5 file not found: {h5_path}")
535        if not video_path.exists():
536            sys.exit(f"[error] Video file not found: {video_path}")
537        if out_path.exists():
538            sys.exit(f"[error] Output file already exists: {out_path}\n"
539                     "        Rename or move it before running again.")
540        align(h5_path, video_path, out_path, trial_num=trial_num)
541        return
542
543    parser = argparse.ArgumentParser(
544        description="Align camera video with ephys membrane-potential recording.",
545        formatter_class=argparse.RawDescriptionHelpFormatter,
546        epilog=__doc__,
547    )
548    parser.add_argument("h5_file",    type=Path, help="HDF5 ephys recording")
549    parser.add_argument("video_file", type=Path, help="Input camera video (.avi)")
550    parser.add_argument(
551        "output_file", type=Path, nargs="?", default=None,
552        help="Output composite video (default: <h5_stem>_aligned.avi next to HDF5)",
553    )
554    parser.add_argument(
555        "--trial", type=int, default=None, metavar="N",
556        help="Trial number (1-based) to extract from a trial-mode HDF5",
557    )

def find_frame_samples(ttl: numpy.ndarray, threshold: float = 2.5) -> numpy.ndarray: 
100def find_frame_samples(ttl: np.ndarray, threshold: float = TTL_THRESHOLD) -> np.ndarray:
101    """Return sample indices of rising edges in the TTL trigger signal.
102
103    Each rising edge marks the moment the DAQ fired a trigger pulse to
104    the camera, so edge i corresponds to the start of video frame i.
105    """
106    above = (ttl >= threshold).astype(np.int8)
107    diff  = np.diff(above, prepend=np.int8(0))
108    return np.where(diff == 1)[0]

288    # --- load ephys --------------------------------------------------------
289    if _is_trial_mode(h5_path):

294        vm, ttl, sr = _load_trial(h5_path, trial_num)
295    else:

308

361
362    print()  # newline after progress
363
364    cap.release()
365    writer.release()
366    print(f"[done] Saved → {output_path}")
